```typescript
// backend/src/middleware/performance-monitor.ts
import { Request, Response, NextFunction } from 'express';

interface PerformanceMetrics {
  [endpoint: string]: {
    count: number;
    totalTime: number;
    maxTime: number;
    p95: number;
  };
}

const metrics: PerformanceMetrics = {};

export function performanceMonitor(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  const endpoint = `${req.method} ${req.path}`;
  
  // 监听响应完成
  res.on('finish', () => {
    const duration = Date.now() - start;
    
    if (!metrics[endpoint]) {
      metrics[endpoint] = { count: 0, totalTime: 0, maxTime: 0, p95: 0 };
    }
    
    const metric = metrics[endpoint];
    metric.count++;
    metric.totalTime += duration;
    metric.maxTime = Math.max(metric.maxTime, duration);
    
    // 简单估算 P95（实际应使用完整统计）
    if (metric.count === 1 || duration > metric.p95) {
      metric.p95 = duration;
    }
    
    // 记录慢请求
    if (duration > 1000) {
      console.warn(`[PERF] Slow request: ${endpoint} took ${duration}ms`);
    }
  });
  
  next();
}
```

## 🔥 二、Redis 缓存实战

### 2.1 缓存策略设计

#### 策略 1: Cache-Aside（旁路缓存）

**适用场景**: Mapping 数据、配置数据

```typescript
// backend/src/cache/scheduling.cache.ts
import { Redis } from 'ioredis';

export class SchedulingCache {
  private redis: Redis;
  private readonly TTL_MAPPING = 300; // 5 分钟
  private readonly TTL_OCCUPANCY = 60; // 1 分钟
  
  constructor() {
    this.redis = new Redis({
      host: process.env.REDIS_HOST || 'localhost',
      port: parseInt(process.env.REDIS_PORT || '6379'),
      keyPrefix: 'logix:scheduling:'
    });
  }
  
  /**
   * 获取港口 - 车队映射（带缓存）
   */
  async getPortTruckingMapping(portCode: string, country: string) {
    const cacheKey = `mapping:port:${portCode}:country:${country}`;
    
    // 1. 尝试从缓存读取
    const cached = await this.redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }
    
    // 2. 查询数据库
    const mappingRepo = AppDataSource.getRepository(TruckingPortMapping);
    const mappings = await mappingRepo.find({
      where: { portCode, country, isActive: true }
    });
    
    // 3. 写入缓存
    await this.redis.setex(cacheKey, this.TTL_MAPPING, JSON.stringify(mappings));
    
    return mappings;
  }
  
  /**
   * 获取仓库档期占用（短缓存）
   */
  async getWarehouseOccupancy(warehouseCode: string, date: Date) {
    const dateStr = date.toISOString().split('T')[0];
    const cacheKey = `occupancy:warehouse:${warehouseCode}:date:${dateStr}`;
    
    const cached = await this.redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }
    
    const occupancyRepo = AppDataSource.getRepository(ExtWarehouseDailyOccupancy);
    const occupancy = await occupancyRepo.findOne({
      where: { warehouseCode, date }
    });
    
    await this.redis.setex(cacheKey, this.TTL_OCCUPANCY, JSON.stringify(occupancy));
    
    return occupancy;
  }
  
  /**
   * 清除缓存（当数据变更时）
   */
  async invalidateMappingCache(portCode?: string, country?: string) {
    const pattern = portCode && country
      ? `mapping:port:${portCode}:country:${country}*`
      : 'mapping:*';
    
    const keys = await this.redis.keys(pattern);
    if (keys.length > 0) {
      await this.redis.del(...keys);
      logger.info(`[Cache] Invalidated ${keys.length} mapping cache entries`);
    }
  }
}

// 全局实例
export const schedulingCache = new SchedulingCache();
```

#### 策略 2: Write-Through（写穿缓存）

**适用场景**: 频繁更新的档期数据

```typescript
async decrementWarehouseOccupancy(warehouseCode: string, date: Date) {
  const dateOnly = new Date(date);
  dateOnly.setHours(0, 0, 0, 0);
  
  const repo = AppDataSource.getRepository(ExtWarehouseDailyOccupancy);
  
  // 1. 查询现有记录
  let occupancy = await repo.findOne({
    where: { warehouseCode, date: dateOnly }
  });
  
  if (occupancy) {
    // 2. 更新数据库
    occupancy.plannedCount += 1;
    occupancy.remaining -= 1;
    await repo.save(occupancy);
    
    // 3. 同步更新缓存
    const cacheKey = `occupancy:warehouse:${warehouseCode}:date:${dateOnly.toISOString().split('T')[0]}`;
    await schedulingCache.redis.setex(
      cacheKey,
      60,
      JSON.stringify(occupancy)
    );
  } else {
    // 3. 创建新记录 + 缓存
    const warehouse = await AppDataSource.getRepository(Warehouse).findOne({
      where: { warehouseCode },
      select: ['dailyCapacity']
    });
    const capacity = warehouse?.dailyCapacity ?? 20;
    
    occupancy = await repo.save({
      warehouseCode,
      date: dateOnly,
      plannedCount: 1,
      capacity,
      remaining: capacity - 1
    });
    
    // 同步缓存
    const cacheKey = `occupancy:warehouse:${warehouseCode}:date:${dateOnly.toISOString().split('T')[0]}`;
    await schedulingCache.redis.setex(
      cacheKey,
      60,
      JSON.stringify(occupancy)
    );
  }
  
  logger.info(`[Scheduling] Decremented warehouse occupancy: ${warehouseCode} on ${dateOnly.toISOString()}`);
}
```

```typescript
// backend/src/cache/cache.metrics.ts
export class CacheMetrics {
  private hits = 0;
  private misses = 0;
  
  recordHit() {
    this.hits++;
  }
  
  recordMiss() {
    this.misses++;
  }
  
  getHitRate(): number {
    const total = this.hits + this.misses;
    return total > 0 ? (this.hits / total) * 100 : 0;
  }
  
  printStats() {
    console.log(`=== Cache Metrics ===`);
    console.log(`Hits: ${this.hits}, Misses: ${this.misses}`);
    console.log(`Hit Rate: ${this.getHitRate().toFixed(2)}%`);
  }
}

export const cacheMetrics = new CacheMetrics();

// 在缓存操作中记录
async get(key: string) {
  const value = await redis.get(key);
  if (value) {
    cacheMetrics.recordHit();
  } else {
    cacheMetrics.recordMiss();
  }
  return value;
}
```

### 5.2 性能指标收集

**创建性能仪表板**:
```typescript
// backend/src/utils/performance-tracker.ts
class PerformanceTracker {
  private metrics: Map<string, number[]> = new Map();
  
  record(operation: string, duration: number) {
    if (!this.metrics.has(operation)) {
      this.metrics.set(operation, []);
    }
    this.metrics.get(operation)!.push(duration);
  }
  
  getStats(operation: string) {
    const values = this.metrics.get(operation) || [];
    if (values.length === 0) return null;
    
    const sorted = [...values].sort((a, b) => a - b);
    const avg = values.reduce((a, b) => a + b, 0) / values.length;
    const p95 = sorted[Math.floor(sorted.length * 0.95)];
    const max = Math.max(...sorted);
    
    return { avg, p95, max, count: values.length };
  }
  
  printReport() {
    console.log('\n=== Performance Report ===\n');
    for (const [operation, values] of this.metrics.entries()) {
      const stats = this.getStats(operation);
      if (stats) {
        console.log(`${operation}:`);
        console.log(`  Avg: ${stats.avg.toFixed(0)}ms`);
        console.log(`  P95: ${stats.p95.toFixed(0)}ms`);
        console.log(`  Max: ${stats.max.toFixed(0)}ms`);
        console.log(`  Count: ${stats.count}\n`);
      }
    }
  }
}
```

```typescript
// 重新计算档期
async recalculateOccupancy(warehouseCode: string, startDate: Date, endDate: Date) {
  const scheduledContainers = await this.containerRepo
    .createQueryBuilder('c')
    .leftJoinAndSelect('c.plannedData', 'pd')
    .where('pd.warehouseId = :warehouseCode', { warehouseCode })
    .andWhere('pd.plannedUnloadDate BETWEEN :start AND :end', {
      start: startDate,
      end: endDate
    })
    .getMany();
  
  // 按日期分组
  const byDate = new Map<string, number>();
  for (const c of scheduledContainers) {
    const date = c.plannedData.plannedUnloadDate;
    byDate.set(date, (byDate.get(date) || 0) + 1);
  }
  
  // 更新档期表
  for (const [date, count] of byDate.entries()) {
    await this.updateOccupancy(warehouseCode, new Date(date), count);
  }
}
```

```yaml
      - alert: HighErrorRate
```
