import { IntelligentSchedulingService } from './intelligentScheduling.service';
import { Container } from '../entities/Container';
import { TruckingCompany } from '../entities/TruckingCompany';
import { Warehouse } from '../entities/Warehouse';

describe('IntelligentSchedulingService', () => {
  let service: IntelligentSchedulingService;
  
  beforeEach(() => {
    service = new IntelligentSchedulingService();
  });

  describe('Unload Mode Determination', () => {
    it('should use Live load when trucking company has no yard', async () => {
      // Arrange
      const container = createTestContainer({
        containerNumber: 'TEST001',
        eta: '2026-03-20'
      });
      
      const truckingCompany: Partial<TruckingCompany> = {
        companyCode: 'TRUCK001',
        hasYard: false
      };
      
      const warehouse: Partial<Warehouse> = {
        warehouseCode: 'WH001',
        dailyCapacity: 20
      };

      // Mock methods
      jest.spyOn(service as any, 'selectTruckingCompany')
        .mockResolvedValue(truckingCompany as TruckingCompany);
      jest.spyOn(service as any, 'getCandidateWarehouses')
        .mockResolvedValue([warehouse as Warehouse]);

      // Act
      const result = await (service as any).scheduleSingleContainer(container, {});

      // Assert
      expect(result.success).toBe(true);
      expect(result.plannedData.unloadModePlan).toBe('Live load');
      expect(result.plannedData.plannedPickupDate)
        .toBe(result.plannedData.plannedUnloadDate);
    });

    it('should use Drop off when trucking company has yard', async () => {
      // Arrange
      const container = createTestContainer({
        containerNumber: 'TEST002',
        eta: '2026-03-20'
      });
      
      const truckingCompany: Partial<TruckingCompany> = {
        companyCode: 'TRUCK002',
        hasYard: true,
        dailyCapacity: 15
      };
      
      const warehouse: Partial<Warehouse> = {
        warehouseCode: 'WH002',
        dailyCapacity: 20
      };

      // Mock methods
      jest.spyOn(service as any, 'selectTruckingCompany')
        .mockResolvedValue(truckingCompany as TruckingCompany);
      jest.spyOn(service as any, 'getCandidateWarehouses')
        .mockResolvedValue([warehouse as Warehouse]);

      // Act
      const result = await (service as any).scheduleSingleContainer(container, {});

      // Assert
      expect(result.success).toBe(true);
      expect(result.plannedData.unloadModePlan).toBe('Drop off');
      expect(result.plannedData.plannedPickupDate)
        .toBeLessThan(result.plannedData.plannedUnloadDate);
    });
  });
});

    const result = await (service as any).scheduleSingleContainer(container, {});

    // Arrange
    const container = createTestContainer({

import { isWeekend, skipWeekends, formatDate } from '../date-utils';

describe('Date Utilities', () => {
  describe('isWeekend', () => {
    it('should return true for Saturday', () => {
      const saturday = new Date('2026-03-21');
      expect(isWeekend(saturday)).toBe(true);
    });

    it('should return true for Sunday', () => {
      const sunday = new Date('2026-03-22');
      expect(isWeekend(sunday)).toBe(true);
    });

    it('should return false for weekdays', () => {
      const monday = new Date('2026-03-23');
      const friday = new Date('2026-03-27');
      expect(isWeekend(monday)).toBe(false);
      expect(isWeekend(friday)).toBe(false);
    });
  });

  describe('skipWeekends', () => {
    it('should skip to Monday when given Saturday', () => {
      const saturday = new Date('2026-03-21');
      const result = skipWeekends(saturday);
      expect(result.getDay()).toBe(1); // Monday
      expect(result.toISOString().split('T')[0]).toBe('2026-03-23');
    });

    it('should skip to Monday when given Sunday', () => {
      const sunday = new Date('2026-03-22');
      const result = skipWeekends(sunday);
      expect(result.getDay()).toBe(1); // Monday
      expect(result.toISOString().split('T')[0]).toBe('2026-03-23');
    });

    it('should not change weekday', () => {
      const tuesday = new Date('2026-03-24');
      const result = skipWeekends(tuesday);
      expect(result).toEqual(tuesday);
    });
  });

  describe('formatDate', () => {
    it('should format date as YYYY-MM-DD', () => {
      const date = new Date('2026-03-20T10:30:00Z');
      expect(formatDate(date)).toBe('2026-03-20');
    });

    it('should handle timezone correctly', () => {
      const date = new Date('2026-03-20T23:59:59Z');
      expect(formatDate(date)).toBe('2026-03-20');
    });
  });
});

import axios from 'axios';

const BASE_URL = 'http://localhost:3000/api/v1';

interface TestResult {
  name: string;
  passed: boolean;
  message?: string;
  duration?: number;
  metrics?: Record<string, any>;
}

async function runIntegrationTests() {
  console.log('🧪 Starting Integration Tests...\n');
  
  const results: TestResult[] = [];
  
  // Test 1: API Connectivity
  results.push(await testApiConnectivity());
  
  // Test 2: Get Scheduling Overview
  results.push(await testGetSchedulingOverview());
  
  // Test 3: Single Container Simulation
  results.push(await testSingleContainerSimulation());
  
  // Test 4: Batch Scheduling (Small)
  results.push(await testBatchScheduling(5));
  
  // Test 5: Capacity Constraints
  results.push(await testCapacityConstraints());
  
  // Print Report
  printTestReport(results);
  
  // Exit with error code if any test failed
  const failedTests = results.filter(r => !r.passed).length;
  if (failedTests > 0) {
    console.error(`\n❌ ${failedTests} test(s) failed`);
    process.exit(1);
  } else {
    console.log('\n✅ All tests passed!');
    process.exit(0);
  }
}

// Test 1: API Connectivity
async function testApiConnectivity(): Promise<TestResult> {
  const startTime = Date.now();
  
  try {
    const response = await axios.get(`${BASE_URL}/containers/scheduling-overview`, {
      params: { startDate: '2026-03-01', endDate: '2026-03-31' }
    });
    
    const passed = response.status === 200 && response.data.success === true;
    
    return {
      name: 'API Connectivity',
      passed,
      message: passed ? 'API is accessible' : 'Invalid response format',
      duration: Date.now() - startTime
    };
  } catch (error: any) {
    return {
      name: 'API Connectivity',
      passed: false,
      message: error.response?.data?.message || error.message,
      duration: Date.now() - startTime
    };
  }
}

// Test 2: Get Scheduling Overview
async function testGetSchedulingOverview(): Promise<TestResult> {
  const startTime = Date.now();
  
  try {
    const response = await axios.get(`${BASE_URL}/containers/scheduling-overview`, {
      params: {
        startDate: '2026-03-01',
        endDate: '2026-03-31',
        country: 'CA'
      }
    });
    
    const data = response.data;
    const passed = data.success === true && 
                   typeof data.pendingCount === 'number' &&
                   Array.isArray(data.warehouses);
    
    return {
      name: 'Get Scheduling Overview',
      passed,
      message: passed 
        ? `Pending: ${data.pendingCount}, Warehouses: ${data.warehouses.length}`
        : 'Invalid response structure',
      duration: Date.now() - startTime,
      metrics: { pendingCount: data.pendingCount }
    };
  } catch (error: any) {
    return {
      name: 'Get Scheduling Overview',
      passed: false,
      message: error.response?.data?.message || error.message,
      duration: Date.now() - startTime
    };
  }
}

// Test 3: Single Container Simulation
async function testSingleContainerSimulation(): Promise<TestResult> {
  const startTime = Date.now();
  
  try {
    const response = await axios.post(`${BASE_URL}/scheduling/simulate`, {
      containerNumber: 'TEST_CONTAINER_001'
    });
    
    const data = response.data;
    const passed = data.success === true && 
                   data.results?.[0]?.plannedData !== undefined;
    
    return {
      name: 'Single Container Simulation',
      passed,
      message: passed 
        ? `Mode: ${data.results[0].plannedData.unloadModePlan}`
        : 'Simulation failed',
      duration: Date.now() - startTime,
      metrics: { unloadMode: data.results?.[0]?.plannedData?.unloadModePlan }
    };
  } catch (error: any) {
    return {
      name: 'Single Container Simulation',
      passed: false,
      message: error.response?.data?.message || error.message,
      duration: Date.now() - startTime
    };
  }
}

// Test 4: Batch Scheduling
async function testBatchScheduling(count: number): Promise<TestResult> {
  const startTime = Date.now();
  
  try {
    const response = await axios.post(`${BASE_URL}/scheduling/schedule`, {
      startDate: '2026-03-01',
      endDate: '2026-03-31',
      country: 'CA',
      limit: count
    });
    
    const data = response.data;
    const duration = Date.now() - startTime;
    const successRate = data.successCount / data.total;
    
    const passed = data.success === true && 
                   successRate >= 0.95 &&
                   duration < 5000; // < 5 seconds
    
    return {
      name: `Batch Scheduling (${count} containers)`,
      passed,
      message: passed 
        ? `Success: ${data.successCount}/${data.total}, Duration: ${duration}ms`
        : `Failed: ${data.failedCount} containers`,
      duration,
      metrics: {
        total: data.total,
        successCount: data.successCount,
        successRate,
        avgResponseTime: duration / count
      }
    };
  } catch (error: any) {
    return {
      name: `Batch Scheduling (${count} containers)`,
      passed: false,
      message: error.response?.data?.message || error.message,
      duration: Date.now() - startTime
    };
  }
}

// Test 5: Capacity Constraints
async function testCapacityConstraints(): Promise<TestResult> {
  try {
    // Try to schedule more containers than capacity
    const response = await axios.post(`${BASE_URL}/scheduling/schedule`, {
      startDate: '2026-03-01',
      endDate: '2026-03-01', // Single day
      country: 'CA',
      limit: 100 // Exceed capacity
    });
    
    const data = response.data;
    
    // Should have some failures due to capacity constraints
    const passed = data.failedCount > 0 && 
                   data.results.some((r: any) => 
                     r.message?.includes('产能不足')
                   );
    
    return {
      name: 'Capacity Constraints Validation',
      passed,
      message: passed 
        ? 'Capacity constraints working correctly'
        : 'Capacity constraints not enforced',
      metrics: { failedCount: data.failedCount }
    };
  } catch (error: any) {
    return {
      name: 'Capacity Constraints Validation',
      passed: false,
      message: error.response?.data?.message || error.message
    };
  }
}

// Print Test Report
function printTestReport(results: TestResult[]) {
  console.log('\n=== Test Report ===\n');
  
  for (const result of results) {
    const icon = result.passed ? '✅' : '❌';
    console.log(`${icon} ${result.name}`);
    console.log(`   ${result.message}`);
    if (result.duration) {
      console.log(`   Duration: ${result.duration}ms`);
    }
    if (result.metrics) {
      console.log(`   Metrics:`, JSON.stringify(result.metrics));
    }
    console.log('');
  }
  
  const passed = results.filter(r => r.passed).length;
  const total = results.length;
  const passRate = (passed / total) * 100;
  
  console.log(`Summary: ${passed}/${total} passed (${passRate.toFixed(1)}%)`);
}

// Run tests
runIntegrationTests().catch(console.error);

```javascript
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Trend, Rate } from 'k6/metrics';

// Custom metrics
const schedulingDuration = new Trend('scheduling_duration');
const successRate = new Rate('scheduling_success');

export const options = {
  stages: [
    { duration: '30s', target: 5 },   // Ramp up
    { duration: '1m', target: 10 },   // Stress test
    { duration: '30s', target: 20 },  // Peak load
    { duration: '30s', target: 0 }    // Cool down
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'],
    scheduling_duration: ['avg<400'],
    scheduling_success: ['rate>0.95'],
    checks: ['rate>0.95']
  },
};

export default function () {
  const payload = {
    startDate: '2026-03-01',
    endDate: '2026-03-31',
    country: 'CA',
    limit: 10
  };
  
  const params = {
    headers: { 'Content-Type': 'application/json' },
  };
  
  const res = http.post(
    `${__ENV.BASE_URL || 'http://localhost:3000/api/v1'}/scheduling/schedule`,
    JSON.stringify(payload),
    params
  );
  
  const success = check(res, {
    'status is 200': (r) => r.status === 200,
    'has results': (r) => r.json().results && r.json().results.length > 0,
    'success rate > 90%': (r) => {
      const data = r.json();
      return data.successCount / data.total > 0.9;
    }
  });
  
  if (success) {
    const data = res.json();
    schedulingDuration.add(data.duration || 0);
    successRate.add(1);
  } else {
    successRate.add(0);
  }
  
  sleep(1);
}
```
