### 2.1 免费期截止日计算（关键！）

#### 算法逻辑

```typescript
// demurrage.service.ts L227-230
const n = Math.max(0, freeDays - 1);
const lastFreeDate = freePeriodUsesWorkingDays(freeDaysBasis)
  ? addWorkingDays(startDate, n)  // 工作日模式
  : addDays(startDate, n);        // 自然日模式
```

#### 核心公式

```
last_free_date = start_date + (free_days - 1) 天

其中:
- free_days = 5 → n = 4 (第 1 天是 start_date 当天)
- 工作日模式：跳过周六、周日
- 自然日模式：连续计算
```

#### 示例验证

**场景 1: 自然日 5 天免费期**
```
start_date = 2026-03-20 (周五)
free_days = 5

计算:
n = max(0, 5-1) = 4
last_free_date = 03-20 + 4 天 = 03-24 (周二)

免费期: 03-20, 03-21, 03-22, 03-23, 03-24 (共 5 天)
计费开始: 03-25 (第 6 天)
```

**场景 2: 工作日 5 天免费期**
```
start_date = 2026-03-20 (周五)
free_days = 5, basis = "工作日"

计算:
n = 4
逐日累加工作日:
  03-20 (周五) ✓ → Day 1
  03-21 (周六) ✗ → 跳过
  03-22 (周日) ✗ → 跳过
  03-23 (周一) ✓ → Day 2
  03-24 (周二) ✓ → Day 3
  03-25 (周三) ✓ → Day 4
  03-26 (周四) ✓ → Day 5 (last_free_date)

last_free_date = 2026-03-26
免费期：5 个工作日
```

### 2.2 计费天数计算

#### 算法逻辑

```typescript
// demurrage.service.ts L241-244
const chargeStart = addDays(lastFreeDate, 1);
const chargeDays = chargePeriodUsesWorkingDays(freeDaysBasis)
  ? workingDaysBetween(chargeStart, endDate)  // 工作日计费
  : daysBetween(chargeStart, endDate);        // 自然日计费
```

#### 核心规则

```
charge_start_date = last_free_date + 1 天
charge_days = f(end_date - charge_start_date, basis)

其中:
- 自然日模式：直接计算日期差
- 工作日模式：只计周一至周五
```

#### 示例验证

```

end_date = 2026-03-27 (周五)

charge_start = 03-25
charge_days = 03-25, 03-26, 03-27 = 3 天

工作日模式:
charge_start = 03-25
逐日检查:
  03-25 (周三) ✓
  03-26 (周四) ✓

如果跨越周末:

charge_days = 03-25, 03-26, 03-27, 03-30 = 4 天 (跳过 03-28/29 周末)
```

### 2.3 阶梯费率计算

#### 算法逻辑

```typescript
// demurrage.service.ts L253-277
if (tiers && tiers.length > 0) {
  const sorted = [...tiers].sort((a, b) => a.fromDay - b.fromDay);
  let remainingDays = chargeDays;
  let currentDay = 1;
  
  for (const tier of sorted) {
    if (remainingDays <= 0) break;
    
    const tierToDay = tier.toDay ?? 99999;
    const tierFromDay = Math.max(tier.fromDay, currentDay);
    const tierEndDay = Math.min(tierToDay, currentDay + remainingDays - 1);
    
    if (tierEndDay < tierFromDay) continue;
    
    const daysInTier = tierEndDay - tierFromDay + 1;
    const subtotal = daysInTier * tier.ratePerDay;
    
    tierBreakdown.push({
      fromDay: tierFromDay,
      toDay: tierEndDay,
      days: daysInTier,
      ratePerDay: tier.ratePerDay,
      subtotal
    });
    
    totalAmount += subtotal;
    remainingDays -= daysInTier;
    currentDay = tierEndDay + 1;
  }
}
```

#### 接口 1: 预测卸柜日的滞港费

```typescript
/**
 * 预测在指定卸柜日产生的滞港费
 * @param containerNumber 柜号
 * @param proposedUnloadDate 拟安排的卸柜日
 * @returns 预估的滞港费和详细信息
 */
async predictDemurrageForUnloadDate(
  containerNumber: string,
  proposedUnloadDate: Date
): Promise<{
  lastFreeDate: Date;
  proposedUnloadDate: Date;
  demurrageDays: number;
  demurrageCost: number;
  tierBreakdown: Array<{ fromDay: number; toDay: number; days: number; ratePerDay: number; subtotal: number }>;
  currency: string;
}> {
  // 1. 获取货柜的到港日期和免费期
  const params = await this.getContainerMatchParams(containerNumber);
  const calculationDates = await this.calculateContainerDates(containerNumber);
  
  // 2. 计算免费期截止日
  const demurrageStart = new Date(params.calculationDates.ataDestPort || params.calculationDates.etaDestPort);
  const lastFreeDate = freePeriodUsesWorkingDays(params.freeDaysBasis)
    ? addWorkingDays(demurrageStart, params.freeDays - 1)
    : addDays(demurrageStart, params.freeDays - 1);
  
  // 3. 计算滞港天数
  const demurrageDays = proposedUnloadDate > lastFreeDate
    ? daysBetween(lastFreeDate, proposedUnloadDate)
    : 0;
  
  // 4. 计算滞港费
  const { totalAmount, tierBreakdown } = calculateSingleDemurrage(
    demurrageStart,
    proposedUnloadDate,
    params.freeDays,
    params.ratePerDay,
    params.tiers,
    params.currency,
    params.freeDaysBasis
  );
  
  return {
    lastFreeDate,
    proposedUnloadDate,
    demurrageDays,
    demurrageCost: totalAmount,
    tierBreakdown,
    currency: params.currency
  };
}
```

#### 接口 2: 计算最优卸柜日以最小化滞港费

```typescript
/**
 * 计算最优卸柜日以最小化滞港费
 * @param containerNumber 柜号
 * @param earliestPossibleDate 最早可行卸柜日（受仓库产能约束）
 * @param searchWindowDays 搜索窗口（默认 7 天）
 * @returns 推荐的卸柜日和预期节省
 */
async findOptimalUnloadDateForMinDemurrage(
  containerNumber: string,
  earliestPossibleDate: Date,
  searchWindowDays: number = 7
): Promise<{
  recommendedUnloadDate: Date;
  currentDemurrageCost: number;
  optimizedDemurrageCost: number;
  potentialSavings: number;
  reasoning: string;
} | null> {
  // 1. 获取货柜信息
  const params = await this.getContainerMatchParams(containerNumber);
  const calculationDates = await this.calculateContainerDates(containerNumber);
  
  // 2. 如果没有到港日期，无法计算
  const demurrageStart = new Date(
    calculationDates.ataDestPort || 
    calculationDates.etaDestPort
  );
  if (!demurrageStart) {
    logger.warn(`[Demurrage] Cannot optimize: missing arrival date for ${containerNumber}`);
    return null;
  }
  
  // 3. 计算免费期截止日
  const lastFreeDate = freePeriodUsesWorkingDays(params.freeDaysBasis)
    ? addWorkingDays(demurrageStart, params.freeDays - 1)
    : addDays(demurrageStart, params.freeDays - 1);
  
  // 4. 评估每个候选卸柜日
  const options = [];
  for (let offset = 0; offset < searchWindowDays; offset++) {
    const candidateDate = addDays(earliestPossibleDate, offset);
    
    // 跳过周末（如果配置了）
    if (this.isWeekend(candidateDate) && this.shouldSkipWeekends()) {
      continue;
    }
    
    // 计算该日期的滞港费
    const { totalAmount } = calculateSingleDemurrage(
      demurrageStart,
      candidateDate,
      params.freeDays,
      params.ratePerDay,
      params.tiers,
      params.currency,
      params.freeDaysBasis
    );
    
    options.push({
      date: candidateDate,
      demurrageCost: totalAmount,
      isWithinFreePeriod: candidateDate <= lastFreeDate
    });
  }
  
  // 5. 选择成本最低的
  const bestOption = options.sort((a, b) => a.demurrageCost - b.demurrageCost)[0];
  const currentOption = options.find(o => o.date === earliestPossibleDate);
  
  if (!currentOption || !bestOption) {
    return null;
  }
  
  // 6. 生成推荐理由
  let reasoning = '';
  if (bestOption.isWithinFreePeriod) {
    reasoning = `推荐安排在免费期内 (${bestOption.date.toISOString().split('T')[0]}), 可避免滞港费`;
  } else if (bestOption.demurrageCost < currentOption.demurrageCost) {
    reasoning = `提前安排可减少滞港费 $${currentOption.demurrageCost - bestOption.demurrageCost}`;
  } else {
    reasoning = `已是最优方案，无法进一步节省`;
  }
  
  return {
    recommendedUnloadDate: bestOption.date,
    currentDemurrageCost: currentOption.demurrageCost,
    optimizedDemurrageCost: bestOption.demurrageCost,
    potentialSavings: currentOption.demurrageCost - bestOption.demurrageCost,
    reasoning
  };
}
```

#### 接口 3: Drop off vs Direct 成本对比

```typescript
/**
 * 比较 Direct Unload 和 Drop off + External Storage 的成本
 * @param containerNumber 柜号
 * @param directUnloadDate Direct 方案的卸柜日
 * @param externalStorageDays Drop off 方案的外部堆场存放天数
 * @returns 成本对比和建议
 */
async compareDirectVsDropOff(
  containerNumber: string,
  directUnloadDate: Date,
  externalStorageDays: number = 5
): Promise<{
  directOption: {
    unloadDate: Date;
    demurrageCost: number;
    transportCost: number;
    totalCost: number;
  };
  dropOffOption: {
    pickupDate: Date;
    unloadDate: Date;
    storageCost: number;
    extraTransportCost: number;
    totalCost: number;
  };
  recommendation: 'DIRECT' | 'DROP_OFF';
  savings: number;
  reasoning: string;
} | null> {
  // 1. 计算 Direct 方案成本
  const directDemurrage = await this.predictDemurrageForUnloadDate(
    containerNumber,
    directUnloadDate
  );
  
  const directTransportCost = await this.estimateTransportCost(
    containerNumber,
    'Live load'
  );
  
  const directTotal = directDemurrage.demurrageCost + directTransportCost;
  
  // 2. 计算 Drop off 方案成本
  const params = await this.getContainerMatchParams(containerNumber);
  const calculationDates = await this.calculateContainerDates(containerNumber);
  
  // Drop off: 提柜后先送到外部堆场
  const pickupDate = new Date(calculationDates.plannedPickupDate || directDemurrage.lastFreeDate);
  const externalUnloadDate = addDays(pickupDate, externalStorageDays);
  
  // 外部堆场费（假设$50/天）
  const storageCost = 50 * externalStorageDays;
  
  // 额外运输费（港口→堆场→仓库）
  const extraTransportCost = 50; // 简化估算
  
  const dropOffTotal = storageCost + extraTransportCost;
  
  // 3. 比较并推荐
  const savings = directTotal - dropOffTotal;
  const recommendation = savings > 0 ? 'DROP_OFF' : 'DIRECT';
  
  let reasoning = '';
  if (savings > 0) {
    reasoning = `Drop off 方案可节省 $${savings} (${(savings / directTotal * 100).toFixed(1)}%)`;
  } else if (savings < 0) {
    reasoning = `Direct 方案更经济，节省 $${Math.abs(savings)}`;
  } else {
    reasoning = '两种方案成本相当';
  }
  
  return {
    directOption: {
      unloadDate: directUnloadDate,
      demurrageCost: directDemurrage.demurrageCost,
      transportCost: directTransportCost,
      totalCost: directTotal
    },
    dropOffOption: {
      pickupDate,
      unloadDate: externalUnloadDate,
      storageCost,
      extraTransportCost,
      totalCost: dropOffTotal
    },
    recommendation,
    savings: Math.abs(savings),
    reasoning
  };
}
```

```typescript
  const demurragePrediction = await this.demurrageService.predictDemurrageForUnloadDate(
    container.containerNumber,
    plannedUnloadDate
  );
  
  // 4. 记录日志用于追踪
  logger.info(
    `[Scheduling] Container ${container.containerNumber}: ` +
    `Unload on ${plannedUnloadDate.toISOString().split('T')[0]}, ` +
    `Demurrage: $${demurragePrediction.demurrageCost}`
  );
  
  // 5. 如果滞港费过高，尝试优化
  if (demurragePrediction.demurrageCost > 500) { // 阈值$500
    const optimization = await this.demurrageService.findOptimalUnloadDateForMinDemurrage(
      container.containerNumber,
      plannedUnloadDate
    );
    
    if (optimization && optimization.potentialSavings > 100) {
      logger.warn(
        `💡 High demurrage alert! Consider rescheduling to ` +
        `${optimization.recommendedUnloadDate.toISOString().split('T')[0]} ` +
        `to save $${optimization.potentialSavings}`
      );
      
      // 尝试调整卸柜日
      if (await this.isWarehouseAvailable(warehouse, optimization.recommendedUnloadDate)) {
        plannedUnloadDate = optimization.recommendedUnloadDate;
        logger.info(`✓ Rescheduled to optimal date`);
      }
    }
  }
}
```

```typescript
/**
 * 排柜成本优化服务
 * 整合滞港费计算、运输费估算、堆场费评估
 */
@Injectable()
export class SchedulingCostOptimizerService {
  constructor(
    private demurrageService: DemurrageService,
    private warehouseService: WarehouseService,
    private truckingService: TruckingService
  ) {}
  
  /**
   * 评估卸柜选项的总成本
   */
  async evaluateUnloadOption(
    containerNumber: string,
    warehouseCode: string,
    unloadDate: Date,
    strategy: 'Direct' | 'Drop off' | 'Expedited'
  ): Promise<{
    unloadDate: Date;
    warehouseCode: string;
    strategy: string;
    costBreakdown: {
      demurrageCost: number;
      storageCost: number;
      transportCost: number;
      handlingCost: number;
    };
    totalCost: number;
  }> {
    // 1. 滞港费
    const demurrage = await this.demurrageService.predictDemurrageForUnloadDate(
      containerNumber,
      unloadDate
    );
    
    // 2. 堆存费（Drop off 模式）
    let storageCost = 0;
    if (strategy === 'Drop off') {
      const storageDays = this.calculateStorageDays(unloadDate);
      storageCost = 50 * storageDays;
    }
    
    // 3. 运输费
    const transportCost = await this.truckingService.estimateCost(
      containerNumber,
      warehouseCode,
      strategy
    );
    
    // 4. 操作费（加急费）
    let handlingCost = 0;
    if (strategy === 'Expedited') {
      handlingCost = 50;
    }
    
    return {
      unloadDate,
      warehouseCode,
      strategy,
      costBreakdown: {
        demurrageCost: demurrage.demurrageCost,
        storageCost,
        transportCost,
        handlingCost
      },
      totalCost: demurrage.demurrageCost + storageCost + transportCost + handlingCost
    };
  }
  
  /**
   * 生成多个候选方案并选择最优
   */
  async generateAndSelectBestOption(
    containerNumber: string,
    warehouses: Warehouse[],
    pickupDate: Date
  ): Promise<{
    selectedWarehouse: Warehouse;
    selectedUnloadDate: Date;
    selectedStrategy: string;
    totalCost: number;
    alternatives: any[];
  }> {
    const options = [];
    
    // 1. Direct 方案（未来 7 天）
    for (const warehouse of warehouses) {
      for (let offset = 0; offset <= 7; offset++) {
        const candidateDate = addDays(pickupDate, offset);
        
        if (!await this.warehouseService.isAvailable(warehouse, candidateDate)) {
          continue;
        }
        
        const option = await this.evaluateUnloadOption(
          containerNumber,
          warehouse.warehouseCode,
          candidateDate,
          'Direct'
        );
        
        options.push(option);
      }
    }
    
    // 2. Drop off 方案
    if (await this.hasDropOffCapability()) {
      const dropOffOption = await this.evaluateDropOffStrategy(containerNumber, pickupDate);
      if (dropOffOption) {
        options.push(dropOffOption);
      }
    }
    
    // 3. 选择成本最低的
    if (options.length === 0) {
      throw new Error('No feasible options found');
    }
    
    const best = options.sort((a, b) => a.totalCost - b.totalCost)[0];
    
    logger.info(
      `[CostOptimizer] Selected best option for ${containerNumber}: ` +
      `${best.strategy} on ${best.unloadDate.toISOString().split('T')[0]}, ` +
      `total cost: $${best.totalCost}`
    );
    
    return {
      selectedWarehouse: await this.warehouseService.findByCode(best.warehouseCode),
      selectedUnloadDate: best.unloadDate,
      selectedStrategy: best.strategy,
      totalCost: best.totalCost,
      alternatives: options.slice(0, 3) // 返回前 3 个备选
    };
  }
}
```
