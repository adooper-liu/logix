### 优先级评估模型

```typescript
function calculatePriority(feedback: Feedback): number {
  const weights = {
    category: {
      bug: 1.0,
      improvement: 0.8,
      feature: 0.6,
      question: 0.3,
    },
    severity: {
      critical: 1.0,
      high: 0.7,
      medium: 0.4,
      low: 0.2,
    },
    frequency: 0.3, // 反馈频率
  };

  const score = weights.category[feedback.category] * 40 + weights.severity[feedback.severity] * 40 + weights.frequency * 20;

  return score; // 0-100
}
```
