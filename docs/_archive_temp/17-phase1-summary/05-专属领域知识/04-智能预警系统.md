## 2. 预警规则引擎 🔥

### 2.1 规则接口定义

```typescript
// backend/src/services/alertService.ts

/**
 * 预警规则接口
 */
interface AlertRule {
  /** 规则ID */
  id: string
  
  /** 规则名称 */
  name: string
  
  /** 规则描述 */
  description: string
  
  /** 风险等级 */
  level: 'warning' | 'danger'
  
  /** 触发条件 */
  condition: (container: Container) => boolean
  
  /** 预警消息生成 */
  message: (container: Container) => string
  
  /** 排序权重（数字越大优先级越高） */
  priority: number
}
```

### 2.2 规则注册中心

```typescript
/**
 * 预警规则注册中心
 */
class AlertRuleRegistry {
  private rules: Map<string, AlertRule> = new Map()
  
  /**
   * 注册规则
   */
  register(rule: AlertRule) {
    this.rules.set(rule.id, rule)
  }
  
  /**
   * 获取所有规则
   */
  getAllRules(): AlertRule[] {
    return Array.from(this.rules.values())
      .sort((a, b) => b.priority - a.priority)
  }
  
  /**
   * 检查货柜是否触发预警
   */
  checkContainer(container: Container): AlertResult[] {
    const results: AlertResult[] = []
    
    for (const rule of this.getAllRules()) {
      if (rule.condition(container)) {
        results.push({
          ruleId: rule.id,
          level: rule.level,
          message: rule.message(container)
        })
      }
    }
    
    return results
  }
}

export const alertRuleRegistry = new AlertRuleRegistry()
```

```typescript
/**
 * 规则1: 最晚提柜日即将超期
 * 触发条件: lastFreeDate <= today + 3 days && lastFreeDate >= today
 */
const RULE_IMMINENT_OVERDUE: AlertRule = {
  id: 'imminent_overdue',
  name: '即将超期',
  description: '最晚提柜日在3天内，即将产生滞港费',
  level: 'warning',
  
  condition: (container: Container) => {
    const today = new Date()
    const threeDaysLater = addDays(today, 3)
    const lastFreeDate = container.lastFreeDate
    
    if (!lastFreeDate) return false
    
    return lastFreeDate <= threeDaysLater && lastFreeDate >= today
  },
  
  message: (container: Container) => {
    const daysLeft = getDaysDiff(container.lastFreeDate, new Date())
    return `最晚提柜日还剩 ${daysLeft} 天，即将产生滞港费`
  },
  
  priority: 10
}
```

```typescript
/**
 * 规则3: 最晚还箱日即将到期

const RULE_IMMINENT_RETURN: AlertRule = {
  id: 'imminent_return',
  name: '即将还箱',
  description: '最晚还箱日在3天内',
  level: 'warning',

    const today = new Date()

  },
  
  message: (container: Container) => {
    const daysLeft = getDaysDiff(container.lastReturnDate, new Date())
    return `最晚还箱日还剩 ${daysLeft} 天`
  },
  
  priority: 5
```

## 4. 预警传播机制

### 4.1 预警级别计算

```typescript
/**
 * 预警级别计算
 * 从多个预警中取最高级别
 */
function calculateAlertLevel(container: Container): AlertLevel {
  const alerts = alertRuleRegistry.checkContainer(container)
  
  if (alerts.length === 0) {
    return 'none'
  }
  
  // Danger > Warning > None
  const hasDanger = alerts.some(a => a.level === 'danger')
  if (hasDanger) return 'danger'
  
  const hasWarning = alerts.some(a => a.level === 'warning')
  if (hasWarning) return 'warning'
  
  return 'none'
}
```

### 4.2 预警信息聚合

```typescript
/**
 * 聚合货柜的预警信息
 */
function aggregateContainerAlerts(container: Container): ContainerAlert {
  const alerts = alertRuleRegistry.checkContainer(container)
  const level = calculateAlertLevel(container)
  
  // 合并所有预警消息
  const messages = alerts.map(a => a.message)
  
  return {
    containerNumber: container.containerNumber,
    level,  // 'none' | 'warning' | 'danger'
    messages,
    hasAlert: alerts.length > 0
  }
}
```

```vue
        <StatCard

        />

  </div>

<script setup lang="ts">
import { computed } from 'vue'

const props = defineProps<{
  containers: Container[]
}>()

const stats = computed(() => {
  let warningCount = 0
  let dangerCount = 0
  let normalCount = 0
  
  for (const container of props.containers) {
    const level = calculateAlertLevel(container)
    if (level === 'danger') dangerCount++
    else if (level === 'warning') warningCount++
    else normalCount++
  }
  
  return {
    total: warningCount + dangerCount,
    warningCount,
    dangerCount,
    normalCount
  }
})
</script>
```

### 6.2 按港口分组统计

```typescript
/**
 * 按港口分组统计预警
 */
function groupAlertsByPort(containers: Container[]): PortAlertStats[] {
  const portMap: Record<string, PortAlertStats> = {}
  
  for (const container of containers) {
    const port = container.destinationPort
    if (!portMap[port]) {
      portMap[port] = {
        portCode: port,
        total: 0,
        warning: 0,
        danger: 0
      }
    }
    
    const level = calculateAlertLevel(container)
    portMap[port].total++
    if (level === 'warning') portMap[port].warning++
    if (level === 'danger') portMap[port].danger++
  }
  
  return Object.values(portMap)
}
```
