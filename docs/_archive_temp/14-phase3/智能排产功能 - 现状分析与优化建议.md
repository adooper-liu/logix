```typescript
async batchSchedule(request: ScheduleRequest): Promise<BatchScheduleResponse> {
  // 1. 查询待排产的货柜 (schedule_status = initial)
  const containers = await this.getContainersToSchedule(request);

  // 2. 按清关可放行日排序（先到先得）
  const sortedContainers = this.sortByClearanceDate(containers);

  const toProcess = sortedContainers.slice(skip, skip + limit);

  }

  // 5. 对每个货柜进行排产

```

**关键发现**:

✅ **优点**:

- 支持分步排产（limit/skip），避免一次性处理过多数据
- 按清关可放行日排序，遵循"先到先得"原则
- 预计算免费日，为排产提供依据

⚠️ **问题**:

- ❌ **性能瓶颈**: 串行处理每个货柜（for 循环），没有并发优化
- ❌ **档期竞争**: 多个货柜可能同时抢占同一档期（无锁机制）

```typescript
private async scheduleSingleContainer(
  container: Container,
  request: ScheduleRequest
): Promise<ScheduleResult> {
  // 1. 获取目的港操作记录
  const destPo = container.portOperations?.find(po => po.portType === 'destination');

  // 2. 计算计划清关日 = ETA（无 ETA 时用 ATA）
  const clearanceDate = destPo.eta || destPo.ata;

  // 3. 计算计划提柜日（若 ETA/ATA 已过，至少为今天）
  let plannedPickupDate = await this.calculatePlannedPickupDate(
    plannedCustomsDate,
    destPo.lastFreeDate
  );

  // 4. 确定候选仓库（根据国家 + 港口映射）
  const warehouses = await this.getCandidateWarehouses(countryCode, portCode);

  // 5. 找最早可用的仓库和卸柜日（从提柜日起查，按仓库日产能）
  const { warehouse, plannedUnloadDate } =
    await this.findEarliestAvailableWarehouse(warehouses, plannedPickupDate);

  // 6. 选择车队（根据仓库 + 港口映射）
  const truckingCompany = await this.selectTruckingCompany(
    warehouse.warehouseCode,
    destPo.portCode
  );

  // 7. 计算计划还箱日
  let lastReturnDate: Date | undefined;
  const emptyReturn = await this.emptyReturnRepo.findOne({ /* ... */ });
  if (emptyReturn?.lastReturnDate) {
    lastReturnDate = new Date(emptyReturn.lastReturnDate);
  } else if (destPo.lastFreeDate) {
    // fallback: 从 LFD + 7 天计算
    lastReturnDate = new Date(destPo.lastFreeDate);
    lastReturnDate.setDate(lastReturnDate.getDate() + 7);
  }

  // 8. 更新数据库（dryRun 模式下跳过）
  if (!_request.dryRun) {
    await this.updateContainerSchedule(container.containerNumber, plannedData);
  }
}
```

#### **2.2 推荐算法实现**

```typescript
interface RecommendationScore {
  costScore: number; // 成本得分（越低越好）
  timeScore: number; // 时间得分（越快越好）
  resourceScore: number; // 资源利用率得分
  customerScore: number; // 客户偏好得分
  totalScore: number; // 加权总分
}

function calculateRecommendation(options: UnloadOption[]): UnloadOption {
  const weights = {
    cost: 0.4,
    time: 0.3,
    resource: 0.2,
    customer: 0.1,
  };

  const scoredOptions = options.map((opt) => {
    const scores: RecommendationScore = {
      costScore: normalizeCost(opt.totalCost),
      timeScore: normalizeTime(opt.unloadDate),
      resourceScore: normalizeResource(opt.warehouse),
      customerScore: normalizeCustomer(opt.warehouse),
      totalScore: 0,
    };

    scores.totalScore = scores.costScore * weights.cost + scores.timeScore * weights.time + scores.resourceScore * weights.resource + scores.customerScore * weights.customer;

    return { option: opt, score: scores };
  });

  // 返回总分最高的方案
  return scoredOptions.sort((a, b) => b.score.totalScore - a.score.totalScore)[0].option;
}
```

**预期收益**: 智能推荐最优方案

```typescript
// 分析历史排产数据，优化推荐准确度
async analyzeHistoricalPatterns(
  warehouseCode: string,
  portCode: string,
  timeframe: 'last_30_days' | 'last_90_days'
): Promise<{
  averageDelayDays: number;
  peakSeasonMonths: number[];
  preferredTruckingCompanies: string[];
}> {
  const historical = await this.getSchedulingHistory(warehouseCode, portCode, timeframe);

  // 分析平均延误天数
  const delays = historical.filter(h => h.actualUnloadDate > h.plannedUnloadDate);
  const avgDelay = delays.reduce((sum, d) => sum + daysBetween(d.plannedUnloadDate, d.actualUnloadDate), 0) / delays.length;

  // 分析旺季月份
  const monthCounts = {};
  historical.forEach(h => {
    const month = new Date(h.plannedUnloadDate).getMonth();
    monthCounts[month] = (monthCounts[month] || 0) + 1;
  });
  const peakMonths = Object.entries(monthCounts)
    .sort(([,a], [,b]) => b - a)
    .slice(0, 3)
    .map(([m]) => parseInt(m));

  // 分析首选车队
  const truckingCounts = {};
  historical.forEach(h => {
    truckingCounts[h.truckingCompanyId] = (truckingCounts[h.truckingCompanyId] || 0) + 1;
  });
  const preferredTrucking = Object.entries(truckingCounts)
    .sort(([,a], [,b]) => b - a)
    .slice(0, 3)
    .map(([id]) => id);

  return {
    averageDelayDays: avgDelay,
    peakSeasonMonths: peakMonths,
    preferredTruckingCompanies: preferredTrucking
  };
}
```
