```typescript
async batchWriteBackComputedDates(options?: {
  limitLastFree?: number
  limitLastReturn?: number
}): Promise<{
  lastFreeWritten: number
  lastReturnWritten: number
  lastFreeProcessed: number
  lastReturnProcessed: number
}> {
  // 1. 获取需要处理的货柜列表
  const lastFreeNumbers = await this.getContainerNumbersWithoutLastFreeDate({
    limit: options?.limitLastFree
  })

  const lastReturnNumbers = await this.getContainerNumbersForLastReturnDate({
    limit: options?.limitLastReturn
  })

  let lastFreeWritten = 0
  let lastReturnWritten = 0

  // 2. 批量处理最晚提柜日
  for (const cn of lastFreeNumbers) {
    try {
      const { result } = await this.calculateForContainer(cn)
      if (!result || !result.calculationDates?.lastPickupDateComputed) continue

      const destPort = await this.portOpRepo.findOne({
        where: { containerNumber: cn },
        order: { portSequence: 'DESC' }
      })

      // ⚠️ 关键：检查是否可以覆盖（不覆盖手工维护的 LFD）
      const canOverwrite = !destPort.lastFreeDateSource || 
                          destPort.lastFreeDateSource === 'computed'

      if (canOverwrite) {
        await this.portOpRepo.update(
          { id: destPort.id },
          {
            lastFreeDate: result.calculationDates.lastPickupDateComputed,
            lastFreeDateMode: result.calculationDates.lastPickupDateMode,
            lastFreeDateSource: 'computed'
          }
        )
        lastFreeWritten++
      }
    } catch (error) {
      logger.warn(`[Demurrage] Failed to update last_free_date for ${cn}:`, error)
    }
  }

  // 3. 批量处理最晚还箱日
  for (const cn of lastReturnNumbers) {
    try {
      const { result } = await this.calculateForContainer(cn)
      if (!result || !result.calculationDates?.lastReturnDateComputed) continue

      let emptyReturn = await this.emptyReturnRepo.findOne({
        where: { containerNumber: cn }
      })

      if (!emptyReturn) {
        // 插入新记录
        emptyReturn = this.emptyReturnRepo.create({
          containerNumber: cn,
          lastReturnDate: result.calculationDates.lastReturnDateComputed
        })
        await this.emptyReturnRepo.save(emptyReturn)
        lastReturnWritten++
      } else if (!emptyReturn.lastReturnDate) {
        // 补写空值
        await this.emptyReturnRepo.update(
          { containerNumber: cn },
          { lastReturnDate: result.calculationDates.lastReturnDateComputed }
        )
        lastReturnWritten++
      }
    } catch (error) {
      logger.warn(`[Demurrage] Failed to update last_return_date for ${cn}:`, error)
    }
  }

  return {
    lastFreeWritten,
    lastReturnWritten,
    lastFreeProcessed: lastFreeNumbers.length,
    lastReturnProcessed: lastReturnNumbers.length
  }
}
```

```typescript
async runSingleContainerFreeDateUpdate(containerNumber: string) {
  // 1. 先进行计算
  const { result, message } = await this.calculateForContainer(containerNumber)

  // ⚠️ 关键区别：单柜更新不自动写回（freeDateWriteMode='none'）
  // 由后续的 writeBackSingleContainerFreeDates 单独处理

  if (!result) {
    return {
      containerNumber,
      updated: false,
      hasResult: false,
      message: message || '缺少计算条件'
    }
  }

  // 2. 获取计算结果中的免费日
  const computedLastFreeDate = result.calculationDates?.lastPickupDateComputed
    ? new Date(result.calculationDates.lastPickupDateComputed)
    : null

  const computedLastReturnDate = result.calculationDates?.lastReturnDateComputed
    ? new Date(result.calculationDates.lastReturnDateComputed)
    : null

  if (!computedLastFreeDate && !computedLastReturnDate) {
    return {
      containerNumber,
      updated: false,
      hasResult: true,
      message: '已匹配标准，但本次未得到可写回的免费日'
    }
  }

  // 3. ⭐ 强制写回（不检查来源，允许覆盖）
  const writeBack = await this.writeBackSingleContainerFreeDates(
    containerNumber,
    computedLastFreeDate,
    computedLastReturnDate,
    result.calculationMode
  )

  const updated = !!(writeBack.lastFreeDateWritten || writeBack.lastReturnDateWritten)
  
  return {
    containerNumber,
    updated,
    hasResult: !!(result || updated),
    message: updated
      ? '更新完成'
      : result
        ? '已计算但无需写回（可能为手工维护的最晚提柜日）'
        : message || '缺少计算条件'
  }
}
```

```typescript
private async writeBackSingleContainerFreeDates(
  containerNumber: string,
  computedLastFreeDate: Date | null,
  computedLastReturnDate: Date | null,
  lastFreeDateMode: 'actual' | 'forecast'
): Promise<{ lastFreeDateWritten: boolean; lastReturnDateWritten: boolean }> {
  let lastFreeDateWritten = false
  let lastReturnDateWritten = false

  const destPort = await this.portOpRepo.findOne({
    where: { containerNumber, portType: 'destination' },
    order: { portSequence: 'DESC' }
  })

  // ⭐ 关键区别：单柜更新允许覆盖所有来源
  if (computedLastFreeDate && destPort) {
    // ✅ 不检查 lastFreeDateSource，直接覆盖
    await this.portOpRepo.update(
      { id: destPort.id },
      {
        lastFreeDate: computedLastFreeDate,
        lastFreeDateMode,
        lastFreeDateSource: 'computed'
      }
    )
    lastFreeDateWritten = true
    
    // 同步到拖车表
    const tt = await this.truckingRepo.findOne({ where: { containerNumber } })
    if (tt) {
      await this.truckingRepo.update(
        { containerNumber },
        { lastPickupDate: computedLastFreeDate }
      )
    }
  }

  // ⭐ 最晚还箱日：允许覆盖旧值
  if (computedLastReturnDate) {
    let emptyReturn = await this.emptyReturnRepo.findOne({
      where: { containerNumber }
    })
    
    if (!emptyReturn) {
      // 插入新记录
      emptyReturn = this.emptyReturnRepo.create({
        containerNumber,
        lastReturnDate: computedLastReturnDate
      })
      await this.emptyReturnRepo.save(emptyReturn)
      lastReturnDateWritten = true
    } else {
      // ✅ 强制覆盖旧值
      await this.emptyReturnRepo.update(
        { containerNumber },
        { lastReturnDate: computedLastReturnDate }
      )
      lastReturnDateWritten = true
    }
  }

  return { lastFreeDateWritten, lastReturnDateWritten }
}
```

```typescript
async execute(): Promise<void> {
  const startTime = Date.now()
  
  try {
    const batchSize = parseInt(process.env.DEMURRAGE_BATCH_SIZE || '200', 10)

    // 1. 批量预计算并保存记录（过去 6 个月的货柜）
    const sixMonthsAgo = new Date()
    sixMonthsAgo.setMonth(sixMonthsAgo.getMonth() - 6)
    const today = new Date()
    const shipmentStart = sixMonthsAgo.toISOString().slice(0, 10)
    const shipmentEnd = today.toISOString().slice(0, 10)

    const computeResult = await this.demurrageService.batchComputeAndSaveRecords({
      shipmentStartDate: shipmentStart,
      shipmentEndDate: shipmentEnd,
      limit: batchSize
    })
    logger.info('[DemurrageWriteBackScheduler] Batch compute records completed', computeResult)

    // 2. 批量写回免费日
    const writeBackResult = await this.demurrageService.runScheduledFreeDateUpdate({
      limitLastFree: Math.floor(batchSize / 2),    // 100 条
      limitLastReturn: Math.floor(batchSize / 2)   // 100 条
    })
    
    const duration = Date.now() - startTime

    logger.info('[DemurrageWriteBackScheduler] Batch write-back completed', {
      ...writeBackResult,
      computeRecords: computeResult,
      duration: `${duration}ms`
    })
  } catch (error) {
    logger.error('[DemurrageWriteBackScheduler] Batch task failed', error)
  } finally {
    this.currentExecution = null
  }
}
```

## 🎯 **选择建议**

### **何时使用批量更新？**

✅ **推荐场景**:
- 系统上线后批量初始化数据
- 定期（如每周）批量维护
- 大量货柜需要更新免费日
- 需要保留手工维护的数据

❌ **不推荐**:
- 只需要更新几个货柜
- 需要强制覆盖手工数据

---

### **何时使用单柜更新？**

✅ **推荐场景**:
- 单个货柜数据异常
- 需要强制覆盖现有数据
- 覆盖手工维护的 LFD/LRD
- 精确修复特定货柜

❌ **不推荐**:
- 大批量更新（效率低）
- 日常维护（工作量大）

---

### **何时依赖定时更新？**

✅ **推荐场景**:
- 日常自动维护
- 保持数据新鲜度
- 补充遗漏的免费日
- 无需人工干预的场景

❌ **不推荐**:
- 紧急数据修复
- 需要立即生效的场景
- 需要强制覆盖的场景

---

## 💡 **最佳实践**

### **组合使用策略**

1. **系统上线初期**:
   - 使用 **批量更新** 初始化所有数据
   - 对异常数据使用 **单柜更新** 修复

2. **日常运维**:
   - 依赖 **定时更新** 自动维护
   - 特殊情况使用 **单柜更新** 精准修复

3. **数据质量保障**:
   - 定期检查定时任务执行情况
   - 对逾期未更新的货柜使用 **批量更新** 补全

---

### **注意事项**

⚠️ **重要提醒**:

1. **批量更新和定时更新不会覆盖手工数据**
   - 如果 LFD 是手工维护的，不会被覆盖
   - 如果需要强制覆盖，使用单柜更新

2. **分批处理避免数据库压力**
   - 批量更新默认每次 500 条
   - 定时更新默认每次 100 条

3. **定时任务频率配置**
   - 默认 6 小时执行一次
   - 可根据业务需求调整

4. **单柜更新会刷新列表**
   - 前端会自动刷新表格数据
   - 注意用户体验
