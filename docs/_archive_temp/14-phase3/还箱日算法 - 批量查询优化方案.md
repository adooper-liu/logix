```typescript
/**
 * 查找车队从 earliestDate 起首个有还箱能力的日期（批量优化版）
 * @param truckingCompanyId 车队 ID
 * @param earliestDate 起始日期
 * @param lastReturnDate 最晚还箱日（可选，若指定则不能超过此日期）
 * @returns 最早可用的还箱日，若找不到则返回 null
 */
private async findEarliestAvailableReturnDate_OPTIMIZED(
  truckingCompanyId: string,
  earliestDate: Date,
  lastReturnDate?: Date
): Promise<Date | null> {
  const startTime = Date.now();

  try {
    // 计算搜索范围
    const maxDaysToSearch = lastReturnDate
      ? Math.min(14, Math.ceil((lastReturnDate.getTime() - earliestDate.getTime()) / (1000 * 60 * 60 * 24)))
      : 14;

    // ✅ Step 1: 生成日期范围数组
    const dateRange = [];
    for (let i = 0; i <= maxDaysToSearch; i++) {
      const date = new Date(earliestDate);
      date.setUTCDate(date.getUTCDate() + i);
      date.setUTCHours(0, 0, 0, 0);
      dateRange.push(date);
    }

    // ✅ Step 2: 批量查询所有日期的占用记录（1 次查询）
    const occupancies = await AppDataSource.getRepository(ExtTruckingReturnSlotOccupancy)
      .createQueryBuilder('occupancy')
      .where('occupancy.truckingCompanyId = :truckingCompanyId', { truckingCompanyId })
      .andWhere('occupancy.slotDate IN (:...dates)', { dates: dateRange })
      .getMany();

    // ✅ Step 3: 转为 Map 便于快速查找
    const occupancyMap = new Map(
      occupancies.map(o => [o.slotDate.toISOString(), o])
    );

    // ✅ Step 4: 内存中查找（无数据库开销）
    let firstDateWithDefaultCapacity: Date | null = null;

    for (const date of dateRange) {
      const occupancy = occupancyMap.get(date.toISOString());

      if (!occupancy) {
        // 无占用记录，需要使用车队默认能力
        if (!firstDateWithDefaultCapacity) {
          const trucking = await AppDataSource.getRepository(TruckingCompany).findOne({
            where: { companyCode: truckingCompanyId },
            select: ['dailyReturnCapacity', 'dailyCapacity']
          });

          const capacity = trucking?.dailyReturnCapacity ?? trucking?.dailyCapacity ?? 20;
          if (capacity > 0) {
            firstDateWithDefaultCapacity = date;
          }
        }

        if (firstDateWithDefaultCapacity) {
          return firstDateWithDefaultCapacity;
        }
      } else if (occupancy.plannedCount < occupancy.capacity) {
        // 有剩余能力
        return date;
      }
    }

    // 找不到可用日期
    return null;

  } finally {
    const duration = Date.now() - startTime;
    logger.debug(
      `[Performance] findEarliestAvailableReturnDate_OPTIMIZED: ${duration}ms`,
      {
        truckingCompanyId,
        startDate: earliestDate.toISOString(),
        maxDays: lastReturnDate ? Math.min(14, Math.ceil((lastReturnDate.getTime() - earliestDate.getTime()) / (1000 * 60 * 60 * 24))) : 14
      }
    );
  }
}
```

## ⚠️ 注意事项

### 1. 数据库索引

确保有以下索引以提升批量查询性能：

```sql
-- 复合索引（强烈推荐）
CREATE INDEX idx_trucking_slot_date
ON ext_trucking_return_slot_occupancy(trucking_company_id, slot_date);

-- 如果已有 UNIQUE 约束，则不需要额外索引
```

### 2. 内存使用

批量查询会将 14 条记录加载到内存，但数据量很小（约几 KB），不会造成内存压力。

### 3. 并发控制

批量查询是只读操作，不需要特殊的并发控制。
