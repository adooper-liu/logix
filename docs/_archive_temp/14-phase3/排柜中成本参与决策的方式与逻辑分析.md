```typescript
// scheduling.controller.ts L1539-1597
compareOptions = async (req: Request, res: Response): Promise<void> => {
  try {
    const { containerNumber, options } = req.body;
    
    // 1. 并行评估所有方案
    const comparisons = await Promise.all(
      options.map(async (option: any) => {
        const costBreakdown = await this.costOptimizerService.evaluateTotalCost(option);
        return {
          option,
          costBreakdown,
          rank: 0
        };
      })
    );
    
    // 2. 按总成本排序（升序）
    comparisons.sort((a, b) => a.costBreakdown.totalCost - b.costBreakdown.totalCost);
    
    // 3. 更新排名
    comparisons.forEach((item, index) => {
      item.rank = index + 1;
    });
    
    // 4. 获取推荐方案（成本最低）
    const recommendedOption = comparisons[0];
    const savings = comparisons.length > 1
      ? comparisons[1].costBreakdown.totalCost - recommendedOption.costBreakdown.totalCost
      : 0;
    
    res.json({
      success: true,
      data: {
        containerNumber,
        comparisons,
        recommendedOption: {
          option: recommendedOption.option,
          reason: `总成本最低，比其他方案节省 $${savings.toFixed(2)}`,
          savings
        }
      }
    });
  } catch (error: any) {
    logger.error('[Scheduling] compareOptions error:', error);
    res.status(500).json({
      success: false,
      message: error.message
    });
  }
};
```

### 2.3 方式三：智能推荐最优方案

**API**: `GET /api/v1/scheduling/recommend-option/:containerNumber`

**使用场景**:
- 用户希望系统自动推荐最优方案
- 探索当前日期前后 7 天的成本变化
- 寻找成本最低的卸柜日期和策略

**核心方法**: `suggestOptimalUnloadDate()`

**实现逻辑**:
```typescript
// schedulingCostOptimizer.service.ts L712-904
async suggestOptimalUnloadDate(
  containerNumber: string,
  warehouse: Warehouse,
  truckingCompany: TruckingCompany,
  basePickupDate: Date,
  lastFreeDate?: Date  // 可选参数
): Promise<{
  suggestedPickupDate: Date;
  suggestedStrategy: 'Direct' | 'Drop off' | 'Expedited';
  originalCost: number;
  optimizedCost: number;
  savings: number;
  savingsPercent: number;
  alternatives: Array<{
    pickupDate: Date;
    strategy: 'Direct' | 'Drop off' | 'Expedited';
    totalCost: number;
    savings: number;
  }>;
}> {
  try {
    // 1. 获取货柜信息
    const container = await this.containerRepo.findOne({
      where: { containerNumber },
      relations: ['portOperations']
    });
    
    // 2. 从权威数据源获取免费期（SKILL 原则）
    const demurrageResult = await this.demurrageService.calculateForContainer(containerNumber);
    
    let effectiveLastFreeDate: Date;
    if (demurrageResult.result?.calculationDates?.lastPickupDateComputed) {
      // 优先使用 DB 中的 last_free_date
      effectiveLastFreeDate = new Date(
        demurrageResult.result.calculationDates.lastPickupDateComputed
      );
    } else if (lastFreeDate) {
      effectiveLastFreeDate = lastFreeDate;
    } else {
      effectiveLastFreeDate = dateTimeUtils.addDays(basePickupDate, 7);
    }
    
    // 3. 计算当前方案成本
    const currentOption: UnloadOption = {
      containerNumber,
      warehouse,
      plannedPickupDate: basePickupDate,
      strategy: truckingCompany.hasYard ? 'Drop off' : 'Direct',
      truckingCompany,
      isWithinFreePeriod: basePickupDate <= effectiveLastFreeDate
    };
    
    const currentBreakdown = await this.evaluateTotalCost(currentOption);
    const originalCost = currentBreakdown.totalCost;
    
    // 4. 向前/向后探索最多 7 天
    const searchRange = 7;
    const candidates: Array<{
      pickupDate: Date;
      strategy: 'Direct' | 'Drop off' | 'Expedited';
      totalCost: number;
    }> = [];
    
    for (let offset = -searchRange; offset <= searchRange; offset++) {
      const candidateDate = dateTimeUtils.addDays(basePickupDate, offset);
      
      // 跳过过去和周末
      if (candidateDate < new Date() || 
          (this.isWeekend(candidateDate) && await this.shouldSkipWeekends())) {
        continue;
      }
      
      // 检查仓库档期
      if (!(await this.isWarehouseAvailable(warehouse, candidateDate))) {
        continue;
      }
      
      // 为每个候选日期评估不同策略
      const strategies: Array<'Direct' | 'Drop off' | 'Expedited'> = [
        'Direct',
        ...(truckingCompany.hasYard ? ['Drop off'] : []),
        ...(candidateDate <= effectiveLastFreeDate ? ['Expedited'] : [])
      ];
      
      for (const strategy of strategies) {
        const option: UnloadOption = {
          containerNumber,
          warehouse,
          plannedPickupDate: candidateDate,
          strategy,
          truckingCompany,
          isWithinFreePeriod: candidateDate <= effectiveLastFreeDate
        };
        
        const breakdown = await this.evaluateTotalCost(option);
        
        candidates.push({
          pickupDate: candidateDate,
          strategy,
          totalCost: breakdown.totalCost
        });
      }
    }
    
    // 5. 找到成本最低的方案
    if (candidates.length === 0) {
      // 无候选方案，返回当前方案
      return {
        suggestedPickupDate: basePickupDate,
        suggestedStrategy: truckingCompany.hasYard ? 'Drop off' : 'Direct',
        originalCost,
        optimizedCost: originalCost,
        savings: 0,
        savingsPercent: 0,
        alternatives: [{
          pickupDate: basePickupDate,
          strategy: truckingCompany.hasYard ? 'Drop off' : 'Direct',
          totalCost: originalCost,
          savings: 0
        }]
      };
    }
    
    // 6. 选择成本最低的候选方案
    const optimalCandidate = candidates.reduce((min, curr) =>
      curr.totalCost < min.totalCost ? curr : min
    );
    
    const optimizedCost = optimalCandidate.totalCost;
    const savings = originalCost - optimizedCost;
    const savingsPercent = originalCost > 0 ? (savings / originalCost) * 100 : 0;
    
    // 7. 返回前 3 个最优方案
    const sortedCandidates = candidates
      .sort((a, b) => a.totalCost - b.totalCost)
      .slice(0, 3)
      .map((candidate) => ({
        ...candidate,
        savings: originalCost - candidate.totalCost
      }));
    
    return {
      suggestedPickupDate: optimalCandidate.pickupDate,
      suggestedStrategy: optimalCandidate.strategy,
      originalCost,
      optimizedCost,
      savings,
      savingsPercent,
      alternatives: sortedCandidates
    };
  } catch (error) {
    log.error(`[CostOptimizer] Failed to suggest optimal date:`, error);
    throw error;
  }
}
```

**决策特点**:
- ✅ **主动探索**: 搜索±7 天范围内的所有可能
- ✅ **全局最优**: 找到成本绝对最低点
- ✅ **量化收益**: 明确显示节省金额和百分比
- ✅ **备选方案**: 提供 Top 3 最优选择
- ⚠️ **计算量大**: 需要评估多个候选方案

### 2.4 方式四：排产流程中的成本优化

**集成方式**: 在智能排产过程中自动调用成本优化

**实现示例**:
```typescript
// costOptimization.integration.example.ts L33-101
async evaluateScheduleCost(
  container: Container,
  plannedPickupDate: Date,
  plannedUnloadDate: Date,
  lastFreeDate: Date
): Promise<{
  currentCost: number;
  optimalCost: number;
  potentialSavings: number;
  optimizationAdvice: string;
  currentBreakdown?: CostBreakdown;
  optimalOption?: UnloadOption;
}> {
  try {
    // 1. 评估当前计划的成本
    const currentOption: UnloadOption = {
      containerNumber: container.containerNumber,
      warehouse: {} as any, // 假设已有仓库信息
      plannedUnloadDate: plannedUnloadDate,
      strategy: 'Direct',
      isWithinFreePeriod: plannedUnloadDate <= lastFreeDate
    };
    
    const currentBreakdown = await this.costOptimizerService.evaluateTotalCost(currentOption);
    const currentCost = currentBreakdown.totalCost;
    
    // 2. 生成所有可行方案并评估成本
    const allOptions = await this.costOptimizerService.generateAllFeasibleOptions(
      container,
      plannedPickupDate,
      lastFreeDate
    );
    
    if (allOptions.length === 0) {
      return {
        currentCost,
        optimalCost: currentCost,
        potentialSavings: 0,
        optimizationAdvice: '无其他可行方案',
        currentBreakdown
      };
    }
    
    // 3. 选择最优方案
    const bestResult = await this.costOptimizerService.selectBestOption(allOptions);
    const optimalCost = bestResult.option.totalCost || 0;
    const potentialSavings = currentCost - optimalCost;
    
    // 4. 生成优化建议
    const optimizationAdvice = this.generateOptimizationAdvice(potentialSavings, bestResult.option);
    
    return {
      currentCost,
      optimalCost,
      potentialSavings,
      optimizationAdvice,
      currentBreakdown,
      optimalOption: bestResult.option
    };
  } catch (error) {
    logger.error('[CostOptimization] Error evaluating schedule cost:', error);
    return {
      currentCost: 0,
      optimalCost: 0,
      potentialSavings: 0,
      optimizationAdvice: '成本评估失败'
    };
  }
}
```

- ✅ **嵌入式决策**: 成本优化作为排产流程的一部分

- ✅ **建议生成**: 提供自然语言的优化建议

## 三、成本决策的核心算法

### 3.1 成本构成要素

```typescript
interface CostBreakdown {
  demurrageCost: number;      // 滞港费（免费期外提柜产生）
  detentionCost: number;      // 滞箱费（超期未还箱产生）
  storageCost: number;        // 港口存储费
  yardStorageCost: number;    // 外部堆场堆存费 (Drop off 专属)
  transportationCost: number; // 运输费（基于距离矩阵）
  handlingCost: number;       // 操作费（加急费等）
  totalCost: number;          // 总成本
}
```

**各项成本的计算逻辑**:

| 成本项 | 计算依据 | 影响因素 |
|-------|---------|---------|
| **demurrageCost** | `DemurrageService.calculateTotalCost()` | - 免费期截止日<br>- 提柜日<br>- 滞港费率 |
| **detentionCost** | `DemurrageService.calculateTotalCost()` | - 最晚还箱日<br>- 实际还箱日<br>- 滞箱费率 |
| **storageCost** | 港口存储天数 × 费率 | - 到港日<br>- 卸柜日<br>- 港口存储费率 |
| **yardStorageCost** | 堆场存放天数 × 费率 + 操作费 | - 提柜日<br>- 送仓日<br>- 车队是否有堆场<br>- TruckingPortMapping 费率 |
| **transportationCost** | 距离矩阵 × 基础费率 | - 港口代码<br>- 仓库代码<br>- 距离矩阵 |
| **handlingCost** | 配置项读取 | - `expedited_handling_fee` (默认$50)<br>- 是否为 Expedited 策略 |

### 3.2 方案生成算法

```typescript
async generateAllFeasibleOptions(
  container: Container,
  pickupDate: Date,
  lastFreeDate: Date,
  searchWindowDays: number = 7
): Promise<UnloadOption[]> {
  const options: UnloadOption[] = [];
  
  // 1. 获取候选仓库列表
  const warehouses = await this.getCandidateWarehouses(
    (container as any).countryCode || 'US',
    (container as any).portCode || 'LAX'
  );
  
  // 2. 为每个仓库生成 Direct 方案
  for (const warehouse of warehouses) {
    for (let offset = 0; offset < searchWindowDays; offset++) {
      const candidateDate = dateTimeUtils.addDays(pickupDate, offset);
      
      // 跳过周末（如果配置了）
      if (this.isWeekend(candidateDate) && await this.shouldSkipWeekends()) {
        continue;
      }
      
      // 检查仓库档期
      if (!await this.isWarehouseAvailable(warehouse, candidateDate)) {
        continue;
      }
      
      options.push({
        containerNumber: container.containerNumber,
        warehouse,
        plannedPickupDate: candidateDate,
        strategy: 'Direct',
        isWithinFreePeriod: candidateDate <= lastFreeDate
      });
    }
  }
  
  // 3. 生成 Drop off 方案
  const dropOffOptions = await this.generateDropOffOptions(
    container, 
    pickupDate, 
    lastFreeDate
  );
  options.push(...dropOffOptions);
  
  // 4. 生成 Expedited 方案（仅在紧急情况下）
  const expeditedOptions = await this.generateExpeditedOptions(
    container, 
    lastFreeDate
  );
  options.push(...expeditedOptions);
  
  return options;
}
```

**方案生成策略**:

1. **Direct 方案**: 搜索提柜日起 7 天内的所有可行日期
2. **Drop off 方案**: 仅在免费期外生成，利用堆场缓冲
3. **Expedited 方案**: 仅在免费期即将到期时（≤2 天）生成

### 3.3 最优方案选择算法

```typescript
async selectBestOption(options: UnloadOption[]): Promise<{
  option: UnloadOption;
  costBreakdown: CostBreakdown;
}> {
  if (options.length === 0) {
    throw new Error('No feasible options available');
  }
  
  // 1. 并行评估所有方案的成本
  const evaluatedOptions = await Promise.all(
    options.map(async (option) => ({
      option,
      costBreakdown: await this.evaluateTotalCost(option)
    }))
  );
  
  // 2. 填充 totalCost
  evaluatedOptions.forEach((item) => {
    item.option.totalCost = item.costBreakdown.totalCost;
  });
  
  // 3. 按总成本升序排序
  const best = evaluatedOptions.sort(
    (a, b) => (a.option.totalCost || 0) - (b.option.totalCost || 0)
  )[0];
  
  log.info(
    `[CostOptimizer] Selected optimal option: ` +
    `Strategy=${best.option.strategy}, ` +
    `PickupDate=${best.option.plannedPickupDate.toISOString().split('T')[0]}, ` +
    `Cost=$${best.option.totalCost}`
  );
  
  return best;
}
```

**算法特点**:
- ✅ **简单高效**: 直接按总成本排序
- ✅ **确定性**: 总是返回成本最低的方案
- ✅ **可解释**: 选择理由清晰明了
- ⚠️ **单一目标**: 仅考虑成本，未考虑时间、资源利用率等其他因素

## 四、成本决策的数据流

### 4.1 典型决策流程

```
用户操作
   ↓
前端发起成本优化请求
   ↓
Controller.receive(req.body)
   ↓
Service.suggestOptimalUnloadDate(...)
   ↓
1. 获取货柜数据（含 portOperations）
2. 从 DB 读取 last_free_date（DemurrageService）
3. 评估当前方案成本
4. 生成候选方案（±7 天 × 多种策略）
5. 并行评估所有候选方案成本
```

## 五、成本决策的影响因素

### 5.1 直接影响因素

| 因素 | 影响机制 | 可控性 |
|-----|---------|-------|
| **提柜日** | 决定是否在免费期内，影响滞港费 | ✅ 可通过调整日期优化 |
| **卸柜方式** | Direct vs Drop off vs Expedited | ✅ 可选择不同策略 |
| **仓库选择** | 不同仓库距离不同，运输费不同 | ✅ 可选择不同仓库 |
| **车队选择** | 是否有堆场影响 Drop off 可行性 | ✅ 可选择不同车队 |
| **免费期截止日** | 硬性约束，无法改变 | ❌ 由船公司政策决定 |

### 5.2 间接影响因素

| 因素 | 影响路径 | 可控性 |
|-----|---------|-------|
| **ETA/ATA** | 影响到港时间，间接影响免费期计算 | ❌ 由海运实际情况决定 |
| **仓库档期** | 容量限制可能迫使选择更高成本的日期 | ⚠️ 部分可控（提前规划） |
| **周末配置** | skip_weekends=true 可能跳过低成本日期 | ✅ 可配置 |
| **距离矩阵** | 影响运输费计算 | ⚠️ 相对固定 |

### 5.3 成本敏感性分析

**示例**: 某货柜在不同提柜日的成本对比

```
场景：CNT123456, WH001, 车队有堆场

提柜日        策略         总成本     主要成本构成
------------------------------------------------------------
2026-03-20   Direct       $150      运输费$150
2026-03-21   Direct       $150      运输费$150
2026-03-22   Direct       $150      运输费$150
2026-03-23   Direct       $450      运输费$150 + 滞港费$300 (免费期外)
2026-03-20   Drop off     $400      运输费$150 + 堆场费$250
2026-03-21   Drop off     $400      运输费$150 + 堆场费$250
2026-03-22   Drop off     $700      运输费$150 + 堆场费$250 + 滞港费$300
2026-03-19   Expedited    $200      运输费$150 + 加急费$50
```

**结论**:
- 免费期内 Direct 策略成本最低 ($150)
- 免费期外 Drop off 策略可以避免滞港费，但增加堆场费 ($400 vs $700)
- Expedited 策略用于紧急情况，成本中等 ($200)

```typescript
// 根据业务规则动态调整权重
function calculateDynamicWeights(context: BusinessContext): Weights {
  const weights: Weights = defaultWeights;
  
  // 规则 1: 旺季提高成本权重
  if (context.isPeakSeason) {
    weights.cost += 0.1;
    weights.time -= 0.1;
  }
  
  // 规则 2: VIP 客户提高时间权重
  if (context.customerPriority === 'VIP') {
    weights.time += 0.15;
    weights.cost -= 0.15;
  }
  
  // 规则 3: 紧急订单提高时间权重
  if (context.urgencyLevel === 'HIGH') {
    weights.time += 0.2;
    weights.cost -= 0.2;
  }
  
  // 归一化权重
  const sum = Object.values(weights).reduce((a, b) => a + b, 0);
  return Object.fromEntries(
    Object.entries(weights).map(([k, v]) => [k, v / sum])
  ) as Weights;
}
```
