```typescript
async suggestOptimalUnloadDate(
  containerNumber: string,
  warehouse: Warehouse,
  truckingCompany: TruckingCompany,
  basePickupDate: Date,
  lastFreeDate?: Date
): Promise<{...}> {
  // 1. 获取货柜信息
  const container = await this.containerRepo.findOne({...});
  
  // 2. 获取免费期
  const demurrageResult = await this.demurrageService.calculateForContainer(containerNumber);
  let effectiveLastFreeDate = ...;
  
  // 3. 计算当前方案成本
  const currentOption: UnloadOption = {
    containerNumber,
    warehouse,
    plannedPickupDate: basePickupDate,
    strategy: truckingCompany.hasYard ? 'Drop off' : 'Direct',
    truckingCompany,
    isWithinFreePeriod: basePickupDate <= effectiveLastFreeDate
  };
  
  const currentBreakdown = await this.evaluateTotalCost(currentOption); // ✅ 调用成本评估
  const originalCost = currentBreakdown.totalCost;
  
  // 4. 探索前后 7 天
  for (let offset = -7; offset <= 7; offset++) {
    const candidateDate = dateTimeUtils.addDays(basePickupDate, offset);
    
    // 跳过过去、周末、仓库不可用
    if (candidateDate < new Date()) continue;
    if (this.isWeekend(candidateDate) && await this.shouldSkipWeekends()) continue;
    if (!(await this.isWarehouseAvailable(warehouse, candidateDate))) continue;
    
    // 为每个候选日期评估不同策略
    const strategies = [
      'Direct',
      ...(truckingCompany.hasYard ? ['Drop off'] : []),
      ...(candidateDate <= effectiveLastFreeDate ? ['Expedited'] : [])
    ];
    
    for (const strategy of strategies) {
      const option: UnloadOption = {
        containerNumber,
        warehouse,
        plannedPickupDate: candidateDate,
        strategy,
        truckingCompany,
        isWithinFreePeriod: candidateDate <= effectiveLastFreeDate
      };
      
      const breakdown = await this.evaluateTotalCost(option); // ✅ 调用成本评估
      
      candidates.push({
        pickupDate: candidateDate,
        strategy,
        totalCost: breakdown.totalCost,
        breakdown
      });
    }
  }
  
  // 5. 找到成本最低的方案
  const optimalCandidate = candidates.reduce((min, curr) =>
    curr.totalCost < min.totalCost ? curr : min
  );
  
  return {
    suggestedPickupDate: optimalCandidate.pickupDate,
    suggestedStrategy: optimalCandidate.strategy,
    originalCost,
    optimizedCost: optimalCandidate.totalCost,
    savings: originalCost - optimalCandidate.totalCost,
    ...
  };
}

async batchOptimize(
  containerNumbers: string[],
  basePickupDate: Date,
  lastFreeDate?: Date
): Promise<{
  results: PriorityContainer[];
  summary: {...};
}> {
  // 1. 分类货柜
  const categories = await this.categorizeContainers(containerNumbers);
  
  // 2. 分配优先级
  const priorityQueue = this.assignPriorities(
    categories.withinFreePeriod,
    categories.overdue
  );
  
  // 3. 按优先级优化每个货柜
  const results: PriorityContainer[] = [];
  
  for (const container of priorityQueue) {
    // ✅ 复用单柜优化的逻辑！
    const optimalResult = await this.suggestOptimalUnloadDate(
      container.containerNumber,
      warehouse,
      truckingCompany,
      container.plannedPickupDate,
      container.lastFreeDate
    );
    
    results.push({
      ...container,
      optimizedPickupDate: optimalResult.suggestedPickupDate,
      optimizedCost: optimalResult.optimizedCost,
      savings: optimalResult.savings
    });
  }
  
  return {
    results,
    summary: {...}
  };
}
```
