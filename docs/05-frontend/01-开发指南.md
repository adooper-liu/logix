## 组件规范

### Vue 组件结构

```vue
<script setup lang="ts">
// 1. 类型定义
interface Props {
  title: string
  data?: Container[]
}

// 2. Props 定义
const props = withDefaults(defineProps<Props>(), {
  title: '',
  data: () => []
})

// 3. 事件定义
const emit = defineEmits<{
  (e: 'update', value: string): void
  (e: 'delete', id: string): void
}>()

// 4. 响应式状态
const loading = ref(false)

// 5. 计算属性
const hasData = computed(() => props.data.length > 0)

// 6. 方法
function handleClick() {
  emit('update', 'new value')
}
</script>

<template>
  <div class="container-card">
    <h3>{{ title }}</h3>
    <div v-if="hasData">
      <slot></slot>
    </div>
    <el-empty v-else description="暂无数据" />
  </div>
</template>

<style scoped>
.container-card {
  padding: 16px;
}
</style>
```

## 组合式函数

### 命名规范

```
use + PascalCase
例: useContainerData, useFilterConditions
```

### 示例

```typescript
// composables/useContainerData.ts
import { ref, computed } from 'vue'
import containerService from '@/services/container'

export function useContainerData() {
  const containers = ref<Container[]>([])
  const loading = ref(false)
  const total = ref(0)

  async function fetchContainers(params: QueryParams) {
    loading.value = true
    try {
      const result = await containerService.getList(params)
      containers.value = result.list
      total.value = result.total
    } finally {
      loading.value = false
    }
  }

  return {
    containers,
    loading,
    total,
    fetchContainers
  }
}
```

## API 调用

### Service 层

```typescript
// services/container.ts
import request from '@/utils/request'

export default {
  getList(params: QueryParams) {
    return request.get('/containers', { params })
  },

  getById(id: string) {
    return request.get(`/containers/${id}`)
  },

  create(data: Container) {
    return request.post('/containers', data)
  },

  update(id: string, data: Partial<Container>) {
    return request.patch(`/containers/${id}`, data)
  },

  delete(id: string) {
    return request.delete(`/containers/${id}`)
  }
}
```
