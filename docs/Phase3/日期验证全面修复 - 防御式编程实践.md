## 📚 技术要点

### 1. isValidDate 工具函数

```typescript
/**
 * 验证 Date 对象是否有效
 */
function isValidDate(date: Date | null | undefined): boolean {
  if (!date) return false;
  return date instanceof Date && !isNaN(date.getTime());
}

// 使用示例
if (!isValidDate(plannedUnloadDate)) {
  logger.warn('Invalid unload date');
  return error;
}
```

### 2. isNaN() 的正确使用

```javascript
// isNaN() 用于检测 NaN
isNaN(NaN)        // true
isNaN(123)        // false
isNaN('abc')      // true (会转换为数字)

// 对于 Date 对象
const invalidDate = new Date('invalid');
isNaN(invalidDate.getTime())  // true ✅

// 正确用法
if (!date || isNaN(date.getTime())) {
  // 处理无效日期
}
```

### 3. 防御式编程 vs 性能

```typescript
// ❌ 过度防御（影响性能）
if (!date || isNaN(date.getTime())) { }
if (!date) { }
if (date instanceof Date) { }
if (!isNaN(date.getTime())) { }

// ✅ 适度防御（推荐）
if (!date || isNaN(date.getTime())) {
  // 关键节点验证
}
```
