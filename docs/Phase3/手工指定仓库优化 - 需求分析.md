## 📊 **问题②: 待排产任务数需要按港口过滤**

### **当前实现**

**现状**:

- ✅ 支持按**国家**过滤 (`country` 参数)
- ✅ 支持按**日期范围**过滤 (`startDate`, `endDate`)
- ❌ **不支持按港口**过滤

**SQL 查询逻辑**:

```sql
-- 当前只按国家和日期过滤
WHERE c.schedule_status IN ('initial', 'issued')
  AND EXISTS (
    SELECT 1 FROM process_port_operations po
    WHERE po.container_number = c.container_number
      AND (po.port_type IS NULL OR po.port_type = 'destination' OR po.port_type = 'transit')
      -- 日期范围过滤
      AND (COALESCE(po.ata, po.eta)::date >= $1::date AND ... <= $2::date)
  )
  -- 国家过滤
  AND EXISTS (
    SELECT 1 FROM biz_replenishment_orders ro
    JOIN biz_customers cu ON ro.customer_code = cu.customer_code
    WHERE ro.container_number = c.container_number AND cu.country = $3
  )
  -- 排除已提柜
  AND NOT EXISTS (
    SELECT 1 FROM process_trucking_transport tt
    WHERE tt.container_number = c.container_number AND tt.pickup_date IS NOT NULL
  )
```

```typescript
export interface ScheduleRequest {
  country?: string;
  portCode?: string;  // ← 新增
  startDate?: string;
  endDate?: string;
  forceSchedule?: boolean;
  containerNumbers?: string[];
  limit?: number;
  skip?: number;
  dryRun?: boolean;
  etaBufferDays?: number;
  designatedWarehouseMode?: boolean;
  designatedWarehouseCode?: string;
}

private async getContainersToSchedule(request: ScheduleRequest): Promise<Container[]> {
  const queryBuilder = this.containerRepo.createQueryBuilder('c')
    .leftJoinAndSelect('c.portOperations', 'po')
    .where('c.schedule_status IN (:...statuses)', { statuses: ['initial', 'issued'] });

  // 国家过滤
  if (request.country) {
    queryBuilder.andWhere(`EXISTS (
      SELECT 1 FROM biz_replenishment_orders ro
      JOIN biz_customers cu ON ro.customer_code = cu.customer_code
      WHERE ro.container_number = c.container_number AND cu.country = :country
    )`, { country: request.country });
  }

  // 港口过滤（新增）
  if (request.portCode) {
    queryBuilder.andWhere(`EXISTS (
      SELECT 1 FROM process_port_operations po
      WHERE po.container_number = c.container_number
        AND po.port_code = :portCode
    )`, { portCode: request.portCode });
  }

  // ... 其他过滤条件

  return queryBuilder.getMany();
}
```
