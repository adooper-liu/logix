### 子表 container_number 自动关联

**问题**: 所有过程表都需要 `container_number` 作为外键，但 Excel 中只有一列。

**解决方案**: 在 `transformRow` 函数中自动复制 `container_number` 到所有子表。

```typescript
// useExcelParser.ts
function transformRow(row: Record<string, any>, fieldMappings: FieldMapping[]): Record<string, any> {
  const result: Record<string, any> = {}
  
  // 1. 正常字段映射
  for (const mapping of fieldMappings) {
    const value = getCellValue(row, mapping)
    result[mapping.field] = mapping.transform ? mapping.transform(value, row) : value
  }
  
  // 2. 自动为所有子表复制 container_number
  const containerNumber = result.container_number
  if (containerNumber) {
    const subTables = ['process_trucking_transport', 'process_warehouse_operations', 
                       'process_empty_return', 'process_port_operations']
    
    for (const mapping of fieldMappings) {
      if (subTables.includes(mapping.table) && mapping.field === 'container_number') {
        result[mapping.field] = containerNumber
      }
    }
  }
  
  return result
}
```

**自动拆分逻辑**:

```typescript
// useFileUpload.ts - groupByTable 函数增强
function groupByTable(transformed: Record<string, any>, fieldMappings: any[]) {
  const tables: Record<string, Record<string, any>> = {}
  
  // 第一轮：收集所有有值的字段
  for (const key in transformed) {
    const mapping = fieldMappings.find(m => m.field === key)
    if (mapping && mapping.table) {
      if (!tables[mapping.table]) {
        tables[mapping.table] = {}
      }
      tables[mapping.table][key] = transformed[key]
    }
  }
  
  // 第二轮：确保子表在主表有 container_number 时也被创建
  const mainTableCN = tables.biz_containers?.container_number || 
                      tables.process_sea_freight?.containerNumber
  
  if (mainTableCN) {
    // 处理 process_port_operations - 自动拆分多港经停记录
    const portOps = tables.process_port_operations || {}
    const portRecords: any[] = []
    
    // 起运港 (port_sequence = 1)
    if (transformed['起运港'] || transformed['port_of_loading']) {
      portRecords.push({
        container_number: mainTableCN,
        port_type: 'origin',
        port_code: transformed['port_of_loading'] || transformed['起运港'],
        port_name: transformed['port_of_loading'] || transformed['起运港'],
        port_sequence: 1,
        eta: transformed['起运港 ETA'] || transformed['eta_origin'],
        ata: transformed['起运港 ATA'] || transformed['ata_origin'],
      })
    }
    
    // 途径港/中转港 (port_sequence = 2)
    if (transformed['途径港'] || transformed['中转港']) {
      portRecords.push({
        container_number: mainTableCN,
        port_type: 'transit',
        port_code: transformed['途径港'] || transformed['中转港'],
        port_name: transformed['途径港'] || transformed['中转港'],
        port_sequence: portRecords.length + 1,
        eta: transformed['途径港 ETA'],
        ata: transformed['途径港 ATA'],
      })
    }
    
    // 目的港 (port_sequence = 最后)
    if (transformed['目的港'] || transformed['port_of_discharge']) {
      portRecords.push({
        container_number: mainTableCN,
        port_type: 'destination',
        port_code: transformed['port_of_discharge'] || transformed['目的港'],
        port_name: transformed['port_of_discharge'] || transformed['目的港'],
        port_sequence: portRecords.length + 1,
        eta: transformed['预计到港日期 (ETA)'] || transformed['eta'],
        ata: transformed['实际到港日期 (ATA)'] || transformed['ata'],
        free_storage_days: transformed['免堆期 (天)'],
        free_detention_days: transformed['场内免箱期 (天)'],
        last_free_date: transformed['最后免费日期'],
      })
    }
    
    // 如果有多个港口记录，转换为数组格式
    if (portRecords.length > 0) {
      tables.process_port_operations = portRecords
    }
  }
  
  return { tables }
}
```

## 飞驼数据导入

### 功能描述

从飞驼系统导出的 Excel 文件中提取数据，并自动更新核心时间表。

### 数据特点

1. **状态码驱动**: 通过飞驼状态码（如 STSP、GITM、BDAR 等）触发时间字段更新
2. **自动推断**: 根据状态码自动推断港口类型（origin/transit/destination）
3. **多表更新**: 同时更新 `ext_feituo_status_events`、`process_port_operations`、`process_sea_freight` 等表

### 状态码映射

#### 起运港阶段 (port_type = 'origin')

| 状态码 | 含义 | 更新字段 | 所在表 |
|-------|------|---------|--------|
| STSP | 提空箱 | `gate_out_time` | process_port_operations |
| GITM | 进场 | `gate_in_time` | process_port_operations |
| DUMP | 甩柜 | `dumped_date` | process_port_operations |
| LOBD | 装船 | `shipment_date` | process_sea_freight |
| DLPT | 离港 | `shipment_date` | process_sea_freight |
| STUF | 装箱 | `stuffing_date` | process_port_operations |

#### 目的港阶段 (port_type = 'destination')

| 状态码 | 含义 | 更新字段 | 所在表 |
|-------|------|---------|--------|
| BDAR/ATA | 抵港 | `ata` | process_port_operations |
| POCA | 靠泊 | `ata` | process_port_operations |
| ETA/FETA | 预计到港 | `eta` | process_port_operations |
| DSCH/DISC | 卸船 | `dest_port_unload_date` | process_port_operations |
| GTIN | 进港 | `gate_in_time` | process_port_operations |
| GTOT/STCS | 提柜 | `gate_out_time` | process_port_operations |
| PCAB | 可提货 | `available_time` | process_port_operations |
| RCVE | 还空箱 | `return_time` | process_empty_return |

### 更新流程

```
1. 读取飞驼 Excel
   ↓
2. 解析状态事件
   ↓
3. 筛选 has_occurred=true 的事件
   ↓
4. 对每个状态码:
   a. getCoreFieldName() → 获取字段名
   b. getPortTypeForStatusCode() → 获取港口类型
   c. 查找/创建对应的 port_operations 记录
   d. 更新时间字段
   ↓
5. 特殊处理:
   - RCVE → 更新 process_empty_return
   - LOBD/DLPT → 更新 process_sea_freight
   ↓
6. 保存变更
```

### 2. 数据类型转换

**推荐做法**: 使用统一的转换函数

```typescript
// 布尔值转换
function parseBoolean(value: unknown): boolean | null {
  if (!value) return null
  const str = String(value).trim().toLowerCase()
  return ['是', 'yes', 'true', '1'].includes(str)
}

// 日期转换
function parseDate(value: unknown): string | null {
  if (!value) return null
  // Excel 日期可能是数字或字符串
  if (typeof value === 'number') {
    // Excel 序列号转日期
    const date = new Date(Math.round((value - 25569) * 86400 * 1000))
    return date.toISOString().split('T')[0]
  }
  return String(value).split(' ')[0]  // 去除时间部分
}

// 数字转换
function parseDecimal(value: unknown): number | null {
  if (!value) return null
  const num = parseFloat(String(value).replace(/,/g, ''))
  return isNaN(num) ? null : num
}
```

```typescript
  await parseExcelFile(file, fieldMappings)
} catch (error) {
  if (error.message.includes('没有数据')) {
    ElMessage.error('Excel 文件中没有数据，请检查文件格式')
  } else if (error.message.includes('缺少必填字段')) {
    ElMessage.error(`数据验证失败：${error.message}`)
  } else {
    ElMessage.error('文件解析失败，请重试')
  }
}
```

### 4. 性能优化

**大数据量处理**:
- ✅ 分批上传（batchSize: 50）
- ✅ 显示进度条
- ✅ 后台异步处理
- ✅ 事务提交（保证数据一致性）

**示例**:
```typescript
async function uploadBatchData(
  endpoint: string,
  data: any[],
  batchSize: number = 50
): Promise<ImportResult> {
  const totalRecords = data.length
  let successCount = 0
  let failedCount = 0
  
  for (let i = 0; i < totalRecords; i += batchSize) {
    const batch = data.slice(i, i + batchSize)
    
    try {
      const response = await axios.post(endpoint, { batch })
      const result = response.data
      successCount += result.success
      failedCount += result.failed
      
      // 更新进度
      uploadProgress.value = Math.min(100, 
        Math.round(((i + batch.length) * 100) / totalRecords)
      )
    } catch (error) {
      failedCount += batch.length
    }
  }
  
  return {
    total: totalRecords,
    success: successCount,
    failed: failedCount,
    errors: []
  }
}
```
