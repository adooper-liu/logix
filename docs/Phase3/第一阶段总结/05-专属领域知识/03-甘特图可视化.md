### 2.2 泳道筛选条件

```typescript
// 泳道筛选条件定义

export const GANTT_FILTERS = {
  // 按到港
  arrival: {
    dimension: 'arrival',
    dateField: 'etaDestPort',  // 使用ETA
    statusFilter: (container) => 
      container.currentPortType !== 'transit' &&
      ['shipped', 'in_transit', 'at_port'].includes(container.logisticsStatus)
  },
  
  // 按计划提柜
  plannedPickup: {
    dimension: 'plannedPickup',
    dateField: 'plannedPickupDate',
    statusFilter: (container) => 
      container.plannedPickupDate && 
      !['picked_up', 'unloaded', 'returned_empty'].includes(container.logisticsStatus)
  },
  
  // 按最晚提柜
  lastPickup: {
    dimension: 'lastPickup',
    dateField: 'lastFreeDate',
    statusFilter: (container) => 
      (container.logisticsStatus === 'at_port' || container.ataDestPort) &&
      container.currentPortType !== 'transit' &&
      !container.hasTruckingRecord
  },
  
  // 按最晚还箱
  lastReturn: {
    dimension: 'lastReturn',
    dateField: 'lastReturnDate',
    statusFilter: (container) => 
      container.logisticsStatus === 'picked_up' || 
      (container.hasTruckingRecord && container.logisticsStatus !== 'returned_empty')
  }
}
```

## 3. 日期取值优先级 🔥

### 3.1 按到港泳道日期取值

**5 级优先级**:

```typescript
// backend/src/services/container.service.ts

/**
 * 甘特图日期取值优先级
 * 用于按到港泳道
 */
function getGanttDate(container: Container, portOperations: PortOperation[]): Date | null {
  
  // 第1优先级: 实际到港日 (ATA)
  if (container.ataDestPort) {
    return container.ataDestPort
  }
  
  // 第2优先级: 修正ETA
  if (container.etaCorrection) {
    return container.etaCorrection
  }
  
  // 第3优先级: 预计到港日 (ETA)
  if (container.etaDestPort) {
    return container.etaDestPort
  }
  
  // 第4优先级: 港口操作表中目的港的日期
  const destinationPortOp = portOperations.find(
    po => po.portType === 'destination' && po.portSequence > 0
  )
  if (destinationPortOp) {
    return destinationPortOp.ataDestPort 
      || destinationPortOp.etaCorrection 
      || destinationPortOp.etaDestPort
  }
  
  // 第5优先级: 海运表的ETA (最后fallback)
  if (container.seaFreight?.eta) {
    return container.seaFreight.eta
  }
  
  return null
}
```

## 4. 筛选逻辑 🔥

### 4.1 按到港筛选

```typescript
// 只显示有实际到港或预计到港的货柜
// 排除中转港
const filterByArrival = (containers: Container[]) => {
  return containers.filter(c => 
    // 条件1: 不是中转港
    c.currentPortType !== 'transit' &&
    // 条件2: 有到港日期（ATA/ETA/修正ETA任一）
    (c.ataDestPort || c.etaCorrection || c.etaDestPort)
  )
}
```

### 4.2 按计划提柜筛选

```typescript
// 只统计有计划提柜日且未提柜的货柜
const filterByPlannedPickup = (containers: Container[]) => {
  return containers.filter(c => 
    // 条件1: 有计划提柜日
    c.plannedPickupDate &&
    // 条件2: 未提柜
    !['picked_up', 'unloaded', 'returned_empty'].includes(c.logisticsStatus)
  )
}
```

### 4.3 按最晚提柜筛选

```typescript
// 已到目的港且未提柜，无拖卡运输记录
const filterByLastPickup = (containers: Container[]) => {
  return containers.filter(c => 
    // 条件1: 已到目的港（有ATA或状态为at_port）
    (c.ataDestPort || c.logisticsStatus === 'at_port') &&
    // 条件2: 不是中转港
    c.currentPortType !== 'transit' &&
    // 条件3: 未提柜（无拖卡运输记录或不是已提柜状态）
    (!c.hasTruckingRecord || c.logisticsStatus !== 'picked_up')
  )
}
```

### 4.4 按最晚还箱筛选

```typescript
// 已提柜或有拖卡运输记录，且未还箱
const filterByLastReturn = (containers: Container[]) => {
  return containers.filter(c => 
    // 条件1: 已提柜或有拖卡运输记录
    (c.logisticsStatus === 'picked_up' || c.hasTruckingRecord) &&
    // 条件2: 未还箱
    c.logisticsStatus !== 'returned_empty' &&
    // 条件3: 有最晚还箱日
    c.lastReturnDate
  )
}
```

### 5.2 useGantt 组合式函数

```typescript
// frontend/src/composables/useGantt.ts

export function useGantt() {
  const containers = ref<Container[]>([])
  const loading = ref(false)
  const dateRange = ref<'7d' | '15d' | '30d'>('7d')
  
  // 按港口分组
  const groupedContainers = computed(() => {
    const groups: Record<string, ContainerGroup> = {}
    
    for (const container of containers.value) {
      const portKey = container.destinationPort
      if (!groups[portKey]) {
        groups[portKey] = {
          key: portKey,
          portName: container.latestPortOperation?.portName || portKey,
          containers: []
        }
      }
      groups[portKey].containers.push(container)
    }
    
    return Object.values(groups)
  })
  
  // 获取圆点样式
  const getDotStyle = (container: Container) => {
    const ganttDate = getGanttDate(container)
    if (!ganttDate) return { display: 'none' }
    
    // 计算位置百分比
    const daysDiff = getDaysDiff(ganttDate, startDate)
    const position = (daysDiff / totalDays) * 100
    
    return {
      left: `${position}%`,
      backgroundColor: getStatusColor(container)
    }
  }
  
  return {
    containers,
    loading,
    dateRange,
    groupedContainers,
    getDotStyle,
    // ...
  }
}
```

### 6.2 分批渲染

```typescript
// 大数据量时分批渲染
const BATCH_SIZE = 100

async function loadContainersInBatches() {
  const allContainers = []
  let offset = 0
  
  while (true) {
    const batch = await fetchContainers({
      limit: BATCH_SIZE,
      offset
    })
    
    allContainers.push(...batch)
    
    if (batch.length < BATCH_SIZE) break
    offset += BATCH_SIZE
    
    // 让出主线程，避免卡顿
    await nextTick()
  }
  
  return allContainers
}
```
