## 1. 滞港费概念 🔥

### 1.1 费用类型对比

| 费用类型 | 英文 | 收费阶段 | 简称 |
|----------|------|---------|------|
| **滞港费** | Demurrage | 货物堆放在港口超期 | 堆存费 |
| **滞箱费** | Detention | 集装箱使用超期 | 箱使费 |
| **超期使用费** | Per Diem | 通用超期计费 | - |

### 1.2 时间线图解

```
┌─────────────────────────────────────────────────────────────────────┐
│                    滞港费/滞箱费时间线                               │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  船到港 ──▶ [免堆期] ──▶ 开始计费 ──▶ 提柜 ──▶ [免箱期] ──▶ 还箱  │
│               │                   │                   │            │
│               │    Demurrage      │    Detention       │            │
│               │   (按天计费)       │   (按天计费)       │            │
│               ▼                   ▼                   ▼            │
│          Free Storage        滞港费开始          滞箱费开始          │
│             Days              $100/天             $80/天           │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

### 1.3 公式

```
Demurrage = max(0, 在港天数 - 免堆期) × 日费率
Detention = max(0, 用箱天数 - 免箱期) × 日费率

总费用 = Demurrage + Detention
```

## 2. 计算模型 🔥

### 2.1 费用计算公式

```typescript
/**
 * 滞港费计算模型
 */
interface DemurrageCalculation {
  // 费用项
  charges: DemurrageCharge[]
  
  // 计算结果
  totalAmount: number
  currency: string
  
  // 计算详情
  calculationDetails: {
    startDate: Date
    endDate: Date
    freeDays: number
    chargeableDays: number
    dailyRate: number
  }
}

/**
 * 单项费用
 */
interface DemurrageCharge {
  chargeType: 'demurrage' | 'detention' | 'storage' | 'other'
  chargeName: string
  freeDays: number
  dailyRate: number
  chargeableDays: number
  amount: number
  isChargeable: boolean  // Y=不收, N=收取
}
```

### 2.2 多行费用项处理

```typescript
/**
 * 滞港费标准可能包含多行费用项
 * 每行代表不同的收费项目，需要分别计算后合计
 */
async function calculateMultipleCharges(
  containerNumber: string,
  standards: DemurrageStandard[]
): Promise<DemurrageCalculation> {
  const charges: DemurrageCharge[] = []
  
  for (const standard of standards) {
    // 检查是否收取
    if (standard.isChargeable === 'Y') {
      // 标记为不收取，跳过
      charges.push({
        chargeType: standard.chargeType,
        chargeName: standard.chargeName,
        freeDays: standard.freeDays,
        dailyRate: standard.dailyRate,
        chargeableDays: 0,
        amount: 0,
        isChargeable: true
      })
      continue
    }
    
    // 计算该项费用
    const chargeableDays = calculateChargeableDays(
      startDate,
      endDate,
      standard.freeDays,
      standard.freeDaysBasis  // '自然日' 或 '工作日'
    )
    
    const amount = chargeableDays * standard.dailyRate
    
    charges.push({
      chargeType: standard.chargeType,
      chargeName: standard.chargeName,
      freeDays: standard.freeDays,
      dailyRate: standard.dailyRate,
      chargeableDays,
      amount,
      isChargeable: false
    })
  }
  
  // 合计所有费用
  const totalAmount = charges.reduce((sum, c) => sum + c.amount, 0)
  
  return { charges, totalAmount, /* ... */ }
}
```

## 3. 标准匹配条件 🔥

### 3.1 四个必须同时满足的条件

```typescript
/**
 * 滞港费标准匹配条件
 * 必须同时满足以下四个字段
 */
interface MatchingConditions {
  destinationCountry: string   // 进口国
  destinationPort: string      // 目的港
  shippingCompany: string      // 船公司
  freightForwarder: string     // 货代公司
}

/**
 * 查询匹配的滞港费标准
 */
async function findMatchingStandard(
  conditions: MatchingConditions
): Promise<DemurrageStandard[]> {
  const currentDate = new Date()
  
  return await demurrageStandardRepo
    .createQueryBuilder('standard')
    .where('standard.destination_country = :country', { 
      country: conditions.destinationCountry 
    })
    .andWhere('standard.destination_port = :port', { 
      port: conditions.destinationPort 
    })
    .andWhere('standard.shipping_company = :carrier', { 
      carrier: conditions.shippingCompany 
    })
    .andWhere('standard.freight_forwarder = :forwarder', { 
      forwarder: conditions.freightForwarder 
    })
    // 生效日期条件
    .andWhere('standard.effective_date <= :currentDate', { 
      currentDate 
    })
    .andWhere(
      // 结束日期为空或大于当前日期
      '(standard.expiry_date IS NULL OR standard.expiry_date >= :currentDate)',
      { currentDate }
    )
    .orderBy('standard.sequence_number', 'ASC')  // 优先级
    .getMany()
}
```

### 3.2 匹配字段来源

```sql
-- 匹配条件字段来源

SELECT 
  c.destination_country,    -- 货柜表
  c.destination_port,        -- 货柜表
  sf.carrier,               -- 海运表 (船公司)

WHERE c.container_number = :containerNumber
```

## 4. 日期计算逻辑 🔥

### 4.1 起算日计算

```typescript
/**
 * 计算起算日
 * 根据"计算方式"字段决定
 */
function calculateStartDate(
  container: Container,
  calculationBasis: '按到港' | '按卸船'
): Date | null {
  
  if (calculationBasis === '按到港') {
    // 按到港：使用 ATA 或 ETA
    return container.ataDestPort || container.etaDestPort
  } else {
    // 按卸船：使用卸船时间
    return container.dischargeTime || container.portOperations
      .find(po => po.portType === 'destination')?.dischargeTime
  }
}
```

### 4.2 截止日计算

```typescript
/**
 * 计算截止日
 * 优先使用实际日期，其次使用计划日期
 */
function calculateEndDate(
  container: Container,
  calculationMode: 'actual' | 'forecast'
): Date | null {
  
  if (calculationMode === 'actual') {
    // 实际模式：使用实际还箱日/提柜日
    return container.returnTime 
      || container.actualPickupDate
      || container.portOperations
        .find(po => po.portType === 'destination')?.gateOutTime
  } else {
    // 预测模式：使用计划还箱日/提柜日
    return container.plannedReturnDate
      || container.plannedPickupDate
  }
}
```

### 4.3 免费天数计算

```typescript
/**
 * 计算免费天数
 * 区分"自然日"和"工作日"
 */
function calculateFreeDays(
  startDate: Date,
  freeDays: number,
  freeDaysBasis: '自然日' | '工作日'
): number {
  
  if (freeDaysBasis === '自然日') {
    // 自然日：直接使用天数
    return freeDays
  } else {
    // 工作日：排除周末
    return countWorkdays(startDate, freeDays)
  }
}

/**
 * 计算工作日天数
 */
function countWorkdays(startDate: Date, days: number): number {
  let count = 0
  let current = new Date(startDate)
  
  while (count < days) {
    const dayOfWeek = current.getDay()
    if (dayOfWeek !== 0 && dayOfWeek !== 6) {  // 0=周日, 6=周六
      count++
    }
    current.setDate(current.getDate() + 1)
  }
  
  return count
}
```

## 5. 计算模式区分 🔥

### 5.1 actual vs forecast

| 模式 | 用途 | 使用条件 | 数据来源 |
|------|------|---------|---------|
| **actual** | 实际费用计算 | 物流已完成 | 实际日期 |
| **forecast** | 费用预测 | 物流未完成 | 计划日期 |

### 5.2 预测模式前置条件

```typescript
/**
 * 预测模式前置条件检查
 */
function checkForecastPreconditions(container: Container): {
  canCalculate: boolean
  reason: string | null
  message: string
} {
  // 条件1: 有实际到港 ATA 或卸船日
  const hasActualArrival = container.ataDestPort || container.dischargeTime
  
  // 条件2: 有计划提柜日
  const hasPlannedPickup = container.plannedPickupDate
  
  // 预测模式：未实际到港且未维护计划提柜日
  if (!hasActualArrival && !hasPlannedPickup) {
    return {
      canCalculate: false,
      reason: 'missing_dates',
      message: '预测模式下，未实际到港且未维护计划提柜日，无法计算滞港费'
    }
  }
  
  return { canCalculate: true, reason: null, message: '' }
}
```

### 5.3 预测模式特殊处理

```typescript
/**
 * 预测模式下的费用计算
 * 不写入数据库，仅返回计算结果
 */
async function calculateForecast(
  containerNumber: string
): Promise<DemurrageCalculation> {
  const container = await getContainer(containerNumber)
  
  // 检查前置条件
  const { canCalculate, reason, message } = checkForecastPreconditions(container)
  if (!canCalculate) {
    return {
      charges: [],
      totalAmount: 0,
      reason,
      message,
      // ...
    }
  }
  
  // 使用计划日期计算
  const calculationDates = {
    startDate: container.ataDestPort || container.etaDestPort,
    endDate: container.plannedPickupDate || container.plannedReturnDate,
    // ...
  }
  
  // 计算费用
  const result = await calculateDemurrage(container, calculationDates, 'forecast')
  
  return result
}
```

## 6. 实际代码实现

### 6.1 核心服务代码

```typescript
// backend/src/services/demurrage.service.ts

/**
 * 计算单个货柜的滞港费
 */
async function calculateForContainer(
  containerNumber: string,
  calculationMode: 'actual' | 'forecast' = 'actual'
): Promise<DemurrageCalculationResult> {
  
  // 1. 获取货柜信息
  const container = await containerService.getContainerByNumber(containerNumber)
  if (!container) {
    throw new Error('货柜不存在')
  }
  
  // 2. 查找匹配的滞港费标准
  const standards = await findMatchingStandard({
    destinationCountry: container.destinationCountry,
    destinationPort: container.destinationPort,
    shippingCompany: container.seaFreight?.carrier,
    freightForwarder: container.order?.freightForwarder
  })
  
  if (standards.length === 0) {
    return {
      success: false,
      error: 'NO_MATCHING_STANDARD',
      message: '未找到匹配的滞港费标准',
      charges: [],
      totalAmount: 0
    }
  }
  
  // 3. 计算日期范围
  const calculationDates = getCalculationDates(container, calculationMode)
  
  // 4. 检查预测模式前置条件
  if (calculationMode === 'forecast') {
    const { canCalculate, reason, message } = checkForecastPreconditions(container)
    if (!canCalculate) {
      return {
        success: true,
        reason,
        message,
        charges: [],
        totalAmount: 0,
        calculationDates
      }
    }
  }
  
  // 5. 计算费用
  const result = await calculateMultipleCharges(container, standards, calculationDates)
  
  return {
    success: true,
    charges: result.charges,
    totalAmount: result.totalAmount,
    calculationDates,
    logisticsStatus: calculateLogisticsStatus(container),
    calculationMode
  }
}
```

### 6.2 API 接口

```typescript
// backend/src/controllers/demurrage.controller.ts

/**
 * GET /api/v1/demurrage/calculate/:containerNumber
 * 计算滞港费

@Get('/calculate/:containerNumber')
async calculateDemurrage(
  @Param('containerNumber') containerNumber: string,
  @Query('mode') calculationMode: 'actual' | 'forecast' = 'actual'

      totalAmount: result.totalAmount,
      calculationDates: result.calculationDates,
      logisticsStatusSnapshot: result.logisticsStatus,
      reason: result.reason,
      message: result.message
    }
  }
}
```
