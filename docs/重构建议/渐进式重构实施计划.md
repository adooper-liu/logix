# 渐进式重构实施计划

## 🎯 重构原则

### ✅ 核心原则

1. **渐进式重构**: 小步快跑，不要一次性完成
2. **测试先行**: 确保每一步都有测试覆盖
3. **文档同步**: 及时更新文档
4. **向后兼容**: 保持现有功能正常
5. **可回滚**: 每一步都可以快速回退

---

## 📋 重构策略

### 策略 1: 分阶段实施

```
阶段 1: 提取 Composables (2 天)
  └─ 不改动 UI，只提取逻辑
  └─ 风险最低，收益明显

阶段 2: 拆分子组件 (2 天)
  └─ 基于已提取的 Composables
  └─ 逐步替换 UI 模块

阶段 3: 优化与完善 (1 天)
  └─ 性能优化
  └─ 文档完善
```

#### Step 2: 提取 Composable (2 小时)

```typescript
// composables/useSchedulingFlow.ts
import { ref } from "vue";
import { containerService } from "@/services/container.service";
import { ElMessage, ElMessageBox } from "element-plus";

interface UseSchedulingFlowOptions {
  onLog: (message: string, type: string) => void;
  onProgress: (progress: number) => void;
}

interface ScheduleParams {
  country?: string;
  startDate?: string;
  endDate?: string;
  dryRun?: boolean;
  etaBufferDays?: number;
  portCode?: string;
}

export function useSchedulingFlow(options: UseSchedulingFlowOptions) {
  const { onLog, onProgress } = options;
  const scheduling = ref(false);
  const BATCH_SIZE = 3;

  /**
   * 批量排产（支持分批确认）
   */
  const handleBatchSchedule = async (params: ScheduleParams) => {
    scheduling.value = true;
    const allResults = [];
    let totalSuccess = 0;
    let totalFailed = 0;

    try {
      onLog("开始排产...", "info");

      let skip = 0;
      let hasMore = true;

      while (hasMore) {
        const result = await containerService.batchSchedule({
          ...params,
          limit: BATCH_SIZE,
          skip,
        });

        // 处理结果
        result.results.forEach((r) => {
          allResults.push(r);
          if (r.success) {
            totalSuccess++;
          } else {
            totalFailed++;
          }
        });

        // 询问是否继续
        if (result.hasMore) {
          try {
            await ElMessageBox.confirm(`本批已处理 ${result.results.length} 个，累计成功 ${totalSuccess} / 失败 ${totalFailed}。是否继续？`, "是否继续", {
              confirmButtonText: "继续",
              cancelButtonText: "停止",
              type: "info",
            });
            skip += BATCH_SIZE;
          } catch {
            onLog("用户选择停止", "info");
            break;
          }
        } else {
          hasMore = false;
        }
      }

      onLog(`排产结束：成功 ${totalSuccess}/${totalSuccess + totalFailed}`, totalFailed > 0 ? "error" : "success");

      return {
        success: true,
        results: allResults,
        totalSuccess,
        totalFailed,
      };
    } catch (error: any) {
      onLog(`排产异常：${error.message}`, "error");
      return {
        success: false,
        error: error.message,
      };
    } finally {
      scheduling.value = false;
    }
  };

  return {
    scheduling,
    handleBatchSchedule,
  };
}
```

      country: "US",
      dryRun: false,
    });

    // Assert

        success: true,
        results: [{ containerNumber: "TEST2", success: true }],

    const { handleBatchSchedule } = useSchedulingFlow({
      onLog: mockOnLog,
      onProgress: mockOnProgress,
    });

    // Act
    const result = await handleBatchSchedule({
      country: "US",
      dryRun: false,
    });

    // Assert - 只处理了第一批

    expect(mockOnLog).toHaveBeenCalledWith("用户选择停止", "info");

### Day 2: 提取 useResourceCapacity (3 小时)

#### Step 1: 创建 Composable (1.5 小时)

```typescript
// composables/useResourceCapacity.ts
import { ref, reactive } from "vue";
import api from "@/services/api";

interface CapacityData {
  status: string;
  occupancyRate: number;
  baseCapacity?: number;
  occupied?: number;
}

interface UseResourceCapacityOptions {
  onError?: (error: Error) => void;
}

export function useResourceCapacity(options: UseResourceCapacityOptions = {}) {
  const { onError } = options;
  const capacityCache = reactive(new Map<string, CapacityData>());
  const loading = ref(false);

  /**
   * 获取仓库档期
   */
  const getWarehouseCapacity = async (warehouseCode: string, date: string): Promise<CapacityData> => {
    const cacheKey = `warehouse:${warehouseCode}:${date}`;

    // 检查缓存
    if (capacityCache.has(cacheKey)) {
      return capacityCache.get(cacheKey)!;
    }

    loading.value = true;
    try {
      const response = await api.get("/scheduling/resources/capacity/range", {
        params: {
          resourceType: "warehouse",
          warehouseCode,
          start: date,
          end: date,
        },
      });

      if (response.data.success && response.data.data.length > 0) {
        const data = response.data.data[0];
        const occupancyRate = data.baseCapacity > 0 ? (data.occupied / data.baseCapacity) * 100 : 0;

        let status = "正常";
        if (occupancyRate >= 95) status = "超负荷";
        else if (occupancyRate >= 80) status = "紧张";

        const result = { status, occupancyRate };
        capacityCache.set(cacheKey, result);
        return result;
      }
    } catch (error) {
      console.error("获取仓库档期失败:", error);
      onError?.(error as Error);
    } finally {
      loading.value = false;
    }

    // 降级：返回默认值
    const defaultData = { status: "正常", occupancyRate: 0 };
    capacityCache.set(cacheKey, defaultData);
    return defaultData;
  };

  /**
   * 获取车队档期
   */
  const getTruckingCapacity = async (truckingCompanyId: string, date: string): Promise<CapacityData> => {
    const cacheKey = `trucking:${truckingCompanyId}:${date}`;

    if (capacityCache.has(cacheKey)) {
      return capacityCache.get(cacheKey)!;
    }

    loading.value = true;
    try {
      const response = await api.get("/scheduling/resources/capacity/range", {
        params: {
          resourceType: "trucking",
          truckingCompanyId,
          start: date,
          end: date,
        },
      });

      if (response.data.success && response.data.data.length > 0) {
        const data = response.data.data[0];
        const occupancyRate = data.baseCapacity > 0 ? (data.occupied / data.baseCapacity) * 100 : 0;

        let status = "正常";
        if (occupancyRate >= 95) status = "超负荷";
        else if (occupancyRate >= 80) status = "紧张";

        const result = { status, occupancyRate };
        capacityCache.set(cacheKey, result);
        return result;
      }
    } catch (error) {
      console.error("获取车队档期失败:", error);
      onError?.(error as Error);
    } finally {
      loading.value = false;
    }

    const defaultData = { status: "正常", occupancyRate: 0 };
    capacityCache.set(cacheKey, defaultData);
    return defaultData;
  };

  /**
   * 批量预加载档期数据
   */
  const preloadCapacity = async (
    requests: Array<{
      type: "warehouse" | "trucking";
      id: string;
      date: string;
    }>,
  ) => {
    const MAX_CONCURRENT = 10;
    const chunks = requests.reduce(
      (acc, req, i) => {
        if (i % MAX_CONCURRENT === 0) acc.push([]);
        acc[acc.length - 1].push(req);
        return acc;
      },
      [] as (typeof requests)[],
    );

    for (const chunk of chunks) {
      const promises = chunk.map(async (req) => {
        try {
          if (req.type === "warehouse") {
            await getWarehouseCapacity(req.id, req.date);
          } else {
            await getTruckingCapacity(req.id, req.date);
          }
        } catch (error) {
          console.warn(`[预加载] ${req.type}:${req.id} 失败:`, error);
        }
      });
      await Promise.all(promises);
    }
  };

  return {
    capacityCache,
    loading,
    getWarehouseCapacity,
    getTruckingCapacity,
    preloadCapacity,
  };
}
```

describe("useResourceCapacity", () => {

            baseCapacity: 100,

        params: expect.objectContaining({
          resourceType: "warehouse",
          warehouseCode: "WH001",
        }),
      }),
    );
  });

  it("should use cache for repeated requests", async () => {

        success: true,
