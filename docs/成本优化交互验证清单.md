### 问题 1: 策略选择的触发条件可能不正确

**当前逻辑**:
```typescript
// Line 857
const strategy = this.selectOptimizationStrategy(category);

// Line 1257-1263
private selectOptimizationStrategy(category: ContainerCategory): OptimizationStrategyConfig {
  if (category.category === 'overdue') {
    return this.OPTIMIZATION_STRATEGIES.OVERDUE; // forward 搜索
  } else {
    return this.OPTIMIZATION_STRATEGIES.WITHIN_FREE_PERIOD; // backward 搜索
  }
}
```

**分类逻辑** (Line 594-608):
```typescript
const today = new Date();
today.setHours(0, 0, 0, 0);
const lastFreeOnly = new Date(lastFreeDate);
lastFreeOnly.setHours(0, 0, 0, 0);

const daysUntilFreezeExpires = Math.ceil(
  (lastFreeOnly.getTime() - today.getTime()) / (1000 * 60 * 60 * 24)
);

const category: ContainerCategory = {
  category: daysUntilFreezeExpires >= 0 ? 'within_free_period' : 'overdue',
  remainingDays: daysUntilFreezeExpires,
  plannedPickupDate,
  lastFreeDate
};
```

**场景分析**:

#### 场景 A: 原计划 2026-03-28，免费期 2026-03-27，今天 2026-03-27
```
today = 2026-03-27
lastFreeDate = 2026-03-27
basePickupDate = 2026-03-28

daysUntilFreezeExpires = (2026-03-27 - 2026-03-27) / 1day = 0
category = 'within_free_period' (因为 0 >= 0)
strategy = WITHIN_FREE_PERIOD (backward 搜索)

生成搜索范围 (从 lastFreeDate 往前):
offset = 0:  2026-03-27 → ❌ 过滤 (dateStr < basePickupDateStr)
offset = -1: 2026-03-26 → ❌ 过滤 (dateStr < basePickupDateStr)
offset = -2: 2026-03-25 → ❌ 过滤 (dateStr < basePickupDateStr)
...

结果：所有日期都被过滤了！❌
```

**问题根源**:
虽然 `lastFreeDate < basePickupDate`（免费期早于原计划），说明原计划已经超期了，但是分类逻辑仍然判断为"免费期内"，导致使用了错误的 backward 策略。

**正确的判断逻辑应该是**:
```typescript
// 不仅要看 lastFreeDate 和 today 的关系
// 还要看 lastFreeDate 和 basePickupDate 的关系

if (lastFreeDate < basePickupDate) {
  // 原计划已经超期，应该使用 forward 策略
  strategy = OVERDUE;
} else if (lastFreeDate < today) {
  // 免费期已过，但原计划还没到，也超期了
  strategy = OVERDUE;
} else {
  // 真正的免费期内
  strategy = WITHIN_FREE_PERIOD;
}
```

## 🛠️ 修复方案

### 修复 1: 改进分类逻辑

**当前代码** (Line 594-608):
```typescript
const daysUntilFreezeExpires = Math.ceil(
  (lastFreeOnly.getTime() - today.getTime()) / (1000 * 60 * 60 * 24)
);

const category = {
  category: daysUntilFreezeExpires >= 0 ? 'within_free_period' : 'overdue',
  remainingDays: daysUntilFreezeExpires,
  ...
};
```

**建议修改**:
```typescript
const today = new Date();
today.setHours(0, 0, 0, 0);
const lastFreeOnly = new Date(lastFreeDate);
lastFreeOnly.setHours(0, 0, 0, 0);
const basePickupOnly = new Date(basePickupDate);
basePickupOnly.setHours(0, 0, 0, 0);

// 计算距离免费期过期的天数
const daysUntilFreezeExpires = Math.ceil(
  (lastFreeOnly.getTime() - today.getTime()) / (1000 * 60 * 60 * 24)
);

// ✅ 新增：判断原计划是否已经超期
const isOriginalPlanOverdue = lastFreeOnly < basePickupOnly;

// 分类逻辑：
// 1. 如果原计划已经超期（lastFreeDate < basePickupDate），无论今天如何，都是 overdue
// 2. 否则，根据剩余天数判断
let category: 'within_free_period' | 'overdue';
let remainingDays: number;

if (isOriginalPlanOverdue) {
  // 原计划超期，即使今天还在免费期内，也是 overdue
  category = 'overdue';
  remainingDays = daysUntilFreezeExpires; // 可能是正数，但原计划确实超期了
} else {
  // 原计划未超期，根据剩余天数判断
  category = daysUntilFreezeExpires >= 0 ? 'within_free_period' : 'overdue';
  remainingDays = daysUntilFreezeExpires;
}
```
