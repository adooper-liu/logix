// 在 IntelligentSchedulingService 类中添加:

/**
 * 批量成本优化（Task 8.1.1）
 *
 * ✅ SKILL 原则:
 * - Leverage: 复用 SchedulingCostOptimizerService.suggestOptimalUnloadDate()
 * - Incremental: 分批处理，控制并发
 * - Knowledge: 共享缓存减少 DB 查询
 */
async batchOptimizeContainers(
  containerNumbers: string[],
  options?: { forceRefresh?: boolean }
): Promise<BatchOptimizeResult[]> {
  const startTime = Date.now();
  logger.info(`[BatchOptimizer] Starting for ${containerNumbers.length} containers`);

  // 1. 读取配置
  const config = await this.schedulingConfigRepo.findOne({
    where: { configKey: 'batch_size_limit' }
  });
  const batchSize = parseInt(config?.configValue || '50');

  // 2. 分批处理
  const batches = this.chunkArray(containerNumbers, batchSize);
  const allResults: BatchOptimizeResult[] = [];

  // 3. 构建共享缓存
  const warehouseCache = new Map<string, Warehouse>();
  const truckingCache = new Map<string, TruckingCompany>();

  // 4. 并发处理每个批次
  for (const batch of batches) {
    const batchPromises = batch.map(number =>
      this.optimizeSingleContainer(number, warehouseCache, truckingCache)
    );
    const batchResults = await Promise.all(batchPromises);
    allResults.push(...batchResults.filter(r => r !== null));
  }

  const totalTime = Date.now() - startTime;
  logger.info(`[BatchOptimizer] Completed in ${totalTime}ms`);

  return allResults;
}

/**
 * 优化单个货柜（内部方法）
 */
private async optimizeSingleContainer(
  containerNumber: string,
  warehouseCache: Map<string, Warehouse>,
  truckingCache: Map<string, TruckingCompany>
): Promise<BatchOptimizeResult | null> {
  try {
    // 1. 获取货柜（带关联）
    const container = await this.containerRepo.findOne({
      where: { containerNumber },
      relations: ['warehouseOperation', 'truckingTransport']
    });

    if (!container) return null;

    // 2. 使用缓存查找仓库和车队
    const warehouseCode = container.warehouseOperation?.warehouseId;
    let warehouse = warehouseCache.get(warehouseCode);
    if (!warehouse) {
      warehouse = await this.warehouseRepo.findOne({ where: { warehouseCode } }) || undefined;
      warehouseCache.set(warehouseCode, warehouse);
    }

    const truckingCompanyId = container.truckingTransport?.truckingCompanyId;
    let truckingCompany = truckingCache.get(truckingCompanyId);
    if (!truckingCompany) {
      truckingCompany = await this.truckingCompanyRepo.findOne({
        where: { companyCode: truckingCompanyId }
      }) || undefined;
      truckingCache.set(truckingCompanyId, truckingCompany);
    }

    if (!warehouse || !truckingCompany) return null;

    // 3. 复用现有的成本优化服务
    const plannedPickupDate = container.truckingTransport?.plannedPickupDate
      ? new Date(container.truckingTransport.plannedPickupDate)
      : new Date();

    const optimization = await this.costOptimizerService.suggestOptimalUnloadDate(
      containerNumber,
      warehouse,
      truckingCompany,
      plannedPickupDate
    );

    // 4. 构建结果
    return {
      containerNumber,
      originalCost: optimization.originalCost,
      optimizedCost: optimization.optimizedCost,
      savings: optimization.savings,
      suggestedPickupDate: optimization.suggestedPickupDate?.toISOString().split('T')[0],
      shouldOptimize: optimization.savings > 0
    };

  } catch (error) {
    logger.error(`[BatchOptimizer] Error for ${containerNumber}:`, error);
    return null;
  }
}

/**
 * 数组长分块工具方法
 */
private chunkArray<T>(array: T[], size: number): T[][] {
  const chunks: T[][] = [];
  for (let i = 0; i < array.length; i += size) {
    chunks.push(array.slice(i, i + size));
  }
  return chunks;
}
```

/**
 * 检查日期是否为周末
 */
export function isWeekend(date: Date): boolean {
  const day = date.getDay();
  return day === 0 || day === 6;
}

/**
 * 如果日期为周末，顺延至下周一
 *
 * @param date 要检查的日期
 * @param skipWeekends 是否跳过周末（从配置读取）
 */
export function adjustForWeekend(date: Date, skipWeekends: boolean = true): Date {
  if (!skipWeekends || !isWeekend(date)) {
    return new Date(date);
  }

  const adjusted = new Date(date);
  const day = date.getDay();

  if (day === 6) {
    // 周六 → 下周一
    adjusted.setDate(adjusted.getDate() + 2);
  } else if (day === 0) {
    // 周日 → 下周一
    adjusted.setDate(adjusted.getDate() + 1);
  }

  return adjusted;
}
```

<template>
  <el-table :data="containers">
    <!-- 添加优化按钮列 -->
    <el-table-column label="优化" width="100">
      <template #default="{ row }">
        <el-button type="primary" size="small" @click="handleOptimize(row)" :loading="optimizing.includes(row.containerNumber)">
          {{ optimizing.includes(row.containerNumber) ? "优化中..." : "优化方案" }}
        </el-button>
      </template>

  </el-table>

  <CostOptimizationPanel
    v-model:visible="showOptimizationPanel"
    :container="selectedContainer"

    @apply="handleApplyOptimization"
    @cancel="showOptimizationPanel = false"

<script setup lang="ts">
import CostOptimizationPanel from "./components/CostOptimizationPanel.vue";

const showOptimizationPanel = ref(false);
const selectedContainer = ref<Container | null>(null);
const originalPlan = ref<any>(null);
const optimizedPlan = ref<any>(null);
const optimizing = ref<string[]>([]);

const handleOptimize = async (container: Container) => {
  optimizing.value.push(container.containerNumber);

  try {
    // 调用后端批量优化接口
    const result = await schedulingService.batchOptimizeContainers([container.containerNumber]);

    if (result && result.length > 0) {
      const opt = result[0];

      // 显示优化面板
      selectedContainer.value = container;
      originalPlan.value = {
        cost: opt.originalCost,
        pickupDate: container.truckingTransport?.plannedPickupDate,

        cost: opt.optimizedCost,
        pickupDate: opt.suggestedPickupDate,
        savings: opt.savings,
      };
      showOptimizationPanel.value = true;
    }
  } finally {
    const idx = optimizing.value.indexOf(container.containerNumber);
    if (idx > -1) {
      optimizing.value.splice(idx, 1);
    }
  }
};

const handleApplyOptimization = async () => {
  // 应用优化方案
  await containerService.updateSchedule(selectedContainer.value!.containerNumber, {
    plannedPickupDate: optimizedPlan.value.pickupDate,
  });

  ElMessage.success("优化方案已应用");
  showOptimizationPanel.value = false;
  loadData(); // 刷新列表
};
