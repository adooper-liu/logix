```typescript
export const isWeekend = (date: Date): boolean => {
  const day = date.getDay();
  return day === 0 || day === 6;
};

export const adjustForWeekend = (date: Date, skipWeekends: boolean = true): Date => {
  if (!skipWeekends || !isWeekend(date)) {
    return new Date(date);
  }

  const adjusted = new Date(date);
  const day = date.getDay();

  if (day === 6) {
    // 周六 → 下周一 (+2 天)
    adjusted.setDate(adjusted.getDate() + 2);
  } else if (day === 0) {
    // 周日 → 下周一 (+1 天)
    adjusted.setDate(adjusted.getDate() + 1);
  }

  return adjusted;
};
```

**并发控制**:

```typescript
for (const batch of batches) {
  const batchPromises = batch.map((number) => this.optimizeSingleContainer(number, warehouseCache, truckingCache));
  const batchResults = await Promise.all(batchPromises);
  allResults.push(...batchResults.filter((r) => r !== null));
}
```

