```typescript
/** 获取运输模式标签类型 */
const getTransportModeTagType = (mode: string) => {
  switch (mode) {
    case "STANDARD":
      return "success"; // 绿色
    case "SEA_RAIL":
      return "warning"; // 橙色
    case "FEEDER":
      return ""; // 蓝色
    default:
      return "info";
  }
};

/** 获取运输模式显示文本 */
const getTransportModeLabel = (mode: string) => {
  switch (mode) {
    case "STANDARD":
      return t("container.logisticsPath.transportMode.standard") || "纯海运";
    case "SEA_RAIL":
      return t("container.logisticsPath.transportMode.seaRail") || "海铁联运";
    case "FEEDER":
      return t("container.logisticsPath.transportMode.feeder") || "驳船联运";
    default:
      return mode;
  }
};
```

### 任务 2.6: 动态调整阶段卡片数量

**实现逻辑**:

```typescript
const nodesByStage = computed(() => {
  if (!path.value?.nodes?.length) return [];

  const transportMode = path.value.transportMode;
  const shouldShowStage = (stage: string) => {
    if (!transportMode) return true;

    // 纯海运：不显示铁路和驳船阶段
    if (transportMode === "STANDARD") {
      return !["rail", "feeder"].includes(stage);
    }

    // 海铁联运：显示铁路阶段
    if (transportMode === "SEA_RAIL") {
      return stage !== "feeder";
    }

    // 驳船联运：显示驳船阶段
    if (transportMode === "FEEDER") {
      return stage !== "rail";
    }

    return true;
  };

  // 过滤掉不需要显示的节点
  const shouldShowNode = (node: StatusNode) => {
    // 不显示"未出运 缺数据"节点
    if (node.status === "NOT_SHIPPED" && (node.rawData as { noData?: boolean })?.noData) {
      return false;
    }
    return true;
  };

  // ... 分组逻辑
});
```

## 技术亮点

### 1. 智能阶段过滤

根据运输模式动态过滤阶段卡片：

- **纯海运**: 只显示标准 9 个阶段
- **海铁联运**: 显示铁路段（10 个阶段）
- **驳船联运**: 显示驳船段（10 个阶段）

### 2. 视觉区分

三种运输模式使用不同颜色标识：

- **纯海运**: 绿色标签
- **海铁联运**: 橙色标签
- **驳船联运**: 蓝色标签

### 3. 节点图标

铁路和驳船节点自动显示特殊图标：

- **铁路节点**: 🚂 火车图标 + 蓝色边框
- **驳船节点**: ⛴️ 小船图标 + 绿色边框

### 4. 预计状态标识

所有预计事件都显示黄色标识：

- 节点卡片：黄色左边框 + 淡黄背景 + "预计"标签
- 详情面板："预计"标签

## 注意事项

### 1. 数据来源

`is_estimated` 字段来自：

- 飞驼 API 同步：`s.isEsti === 'Y'`

- **驳船段**: order 4（海运前）

## 常见问题

### Q1: 为什么看不到铁路/驳船阶段？

**A**: 当前数据库中没有包含铁路或驳船事件的货柜。需要导入相应的测试数据。

### Q2: 预计标签为什么不显示？

**A**: 检查数据库中 `ext_container_status_events.is_estimated` 字段是否有值。

### Q3: 运输模式识别准确吗？

**A**: 基于实际发生的状态码识别：

- 有铁路事件 → SEA_RAIL
- 有驳船事件 → FEEDER
- 都没有 → STANDARD

### Q4: 如何禁用火车/小船图标？

**A**: 修改 `.stage-node-rail::before` 和 `.stage-node-feeder::before` 的 `content` 属性为空字符串。
