### 方案1：基于WMS确认状态（推荐）

```typescript
function determineStatus(containerNumber: string): string {
  // 1. 已还箱
  if (hasReturnTime(containerNumber)) return 'returned_empty';

  // 2. 已卸柜（WMS已确认）
  if (hasWmsConfirmed(containerNumber)) return 'unloaded';

  // 3. 已提柜
  if (hasPickupDate(containerNumber)) return 'picked_up';

  // 4. 已到港（包括中转港和目的港）
  if (hasArrivedAtTransit(containerNumber) || hasArrivedAtDestination(containerNumber)) {
    return 'at_port';
  }

  // 5. 在途
  if (hasShipmentDate(containerNumber)) return 'in_transit';

  // 6. 未出运
  return 'not_shipped';
}

function hasWmsConfirmed(containerNumber: string): boolean {
  const operation = await this.warehouseOperationRepository.findOne({
    where: { containerNumber }
  });

  // WMS状态 = "WMS已完成" OR EBS状态 = "已入库" OR 有WMS确认日期
  return operation && (
    operation.wmsStatus === 'WMS已完成' ||
    operation.ebsStatus === '已入库' ||
    operation.wmsConfirmDate !== null
  );
}
```

## 📋 完整的状态判断逻辑

```typescript
interface WarehouseOperationStatus {
  wmsStatus?: string;      // WMS状态
  ebsStatus?: string;      // EBS状态
  wmsConfirmDate?: Date;   // WMS确认日期
  warehouseArrivalDate?: Date; // 仓库到达日期
  unloadDate?: Date;       // 卸柜日期
}

function determineStatus(containerNumber: string): string {
  // 获取所有操作记录
  const dates = await this.getDateInfo(containerNumber);

  // 1. 已还箱（优先级最高）
  if (dates.returnTime) return 'returned_empty';

  // 2. 已卸柜（基于WMS/EBS状态）
  if (isWmsConfirmed(dates.warehouseOperation)) return 'unloaded';

  // 3. 已提柜
  if (dates.pickupDate) return 'picked_up';

  // 4. 已到港（包括中转港和目的港）
  if (dates.hasTransit || dates.hasDestinationAta) return 'at_port';

  // 5. 在途
  if (dates.shipmentDate) return 'in_transit';

  // 6. 未出运
  return 'not_shipped';
}

function isWmsConfirmed(operation: WarehouseOperationStatus): boolean {
  if (!operation) return false;

  // 判断条件（满足任一即可）
  return (
    // 条件1：WMS状态 = "WMS已完成"
    operation.wmsStatus === 'WMS已完成' ||
    // 条件2：EBS状态 = "已入库"
    operation.ebsStatus === '已入库' ||
    // 条件3：有WMS确认日期
    operation.wmsConfirmDate !== null
  );
}
```

## 🔍 数据验证

### 验证SQL：检查应该更新为unloaded的货柜

```sql
-- 检查picked_up状态但WMS已确认的货柜
SELECT
  c.container_number,
  c.logistics_status,
  wo.wms_status,
  wo.ebs_status,
  wo.wms_confirm_date,
  wo.warehouse_arrival_date,
  er.return_time
FROM biz_containers c
LEFT JOIN process_warehouse_operations wo ON c.container_number = wo.container_number
LEFT JOIN process_empty_returns er ON c.container_number = er."containerNumber"
WHERE c.logistics_status = 'picked_up'
  AND (
    wo.wms_status = 'WMS已完成'
    OR wo.ebs_status = '已入库'
    OR wo.wms_confirm_date IS NOT NULL
  )
  AND er.return_time IS NULL
ORDER BY c.container_number
LIMIT 20;
```

### 批量更新SQL

```sql
-- 更新picked_up → unloaded（基于WMS确认）
UPDATE biz_containers c
SET logistics_status = 'unloaded'
WHERE logistics_status = 'picked_up'
  AND EXISTS (
    SELECT 1 FROM process_warehouse_operations wo
    WHERE wo.container_number = c.container_number
      AND (
        wo.wms_status = 'WMS已完成'
        OR wo.ebs_status = '已入库'
        OR wo.wms_confirm_date IS NOT NULL
      )
  )
  AND NOT EXISTS (
    SELECT 1 FROM process_empty_returns er
    WHERE er.container_number = c.container_number
      AND er.return_time IS NOT NULL
  );
```
