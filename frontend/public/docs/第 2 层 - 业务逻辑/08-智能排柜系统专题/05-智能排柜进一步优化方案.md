### 一、短期优化（1-2 周）

#### 1.1 缓存机制

**问题**：滞港费标准、仓库档期等数据频繁查询数据库。

**方案**：引入 Redis 缓存热点数据。

```typescript
// backend/src/services/CacheService.ts

interface CacheOptions {
  ttl: number;        // 过期时间（秒）
  prefix: string;     // 缓存键前缀
}

export class CacheService {
  private redis = AppDataSource.redis; // 假设已配置 Redis

  /**
   * 获取缓存
   */
  async get<T>(key: string): Promise<T | null> {
    const cached = await this.redis.get(key);
    return cached ? JSON.parse(cached) : null;
  }

  /**
   * 设置缓存
   */
  async set(key: string, value: any, ttl: number = 3600): Promise<void> {
    await this.redis.setex(key, ttl, JSON.stringify(value));
  }

  /**
   * 清除缓存
   */
  async invalidate(pattern: string): Promise<void> {
    const keys = await this.redis.keys(pattern);
    if (keys.length > 0) {
      await this.redis.del(...keys);
    }
  }
}

// 使用示例：demurrage.service.ts
export class DemurrageService {
  private cache = new CacheService();

  async getDemurrageStandard(portCode: string, shipper: string, forwarder: string) {
    const cacheKey = `demurrage:${portCode}:${shipper}:${forwarder}`;

    // 先查缓存
    const cached = await this.cache.get<DemurrageStandard>(cacheKey);
    if (cached) return cached;

    // 缓存未命中，查数据库
    const standard = await this.standardRepo.findOne({ where: { portCode, shipper, forwarder } });

    // 写入缓存（1 小时过期）
    if (standard) {
      await this.cache.set(cacheKey, standard, 3600);
    }

    return standard;
  }
}
```

**预期效果**：滞港费标准查询从 10ms 降至 <1ms。

#### 1.2 批量查询优化

**问题**：当前逐日查询仓库档期，100 柜 × 15 天 = 1500 次查询。

**方案**：一次性获取未来 30 天档期。

```typescript
// backend/src/services/OccupancyCalculator.ts

export class OccupancyCalculator {
  /**
   * 批量获取仓库档期（优化版）
   *
   * @param warehouseCodes 仓库代码列表
   * @param startDate 开始日期
   * @param days 天数
   */
  async getBatchWarehouseOccupancy(
    warehouseCodes: string[],
    startDate: Date,
    days: number = 30
  ): Promise<Map<string, Map<string, WarehouseOccupancy>>> {
    const endDate = dateTimeUtils.addDays(startDate, days);

    // 一次性查询所有仓库的未来档期
    const occupancies = await this.occupancyRepo
      .createQueryBuilder('occ')
      .where('occ.warehouse_code IN (:...warehouseCodes)', { warehouseCodes })
      .andWhere('occ.date >= :startDate', { startDate })
      .andWhere('occ.date <= :endDate', { endDate })
      .getMany();

    // 按 仓库+日期 组织
    const result = new Map<string, Map<string, WarehouseOccupancy>>();

    for (const occ of occupancies) {
      if (!result.has(occ.warehouseCode)) {
        result.set(occ.warehouseCode, new Map());
      }
      result.get(occ.warehouseCode)!.set(
        dateTimeUtils.formatDate(occ.date),
        occ
      );
    }

    return result;
  }

  /**
   * 使用批量档期快速检查可用性
   */
  async findEarliestAvailableDayWithBatch(
    warehouseCode: string,
    startDate: Date,
    batchOccupancy: Map<string, Map<string, WarehouseOccupancy>>,
    maxDays: number = 30
  ): Promise<Date | null> {
    const warehouseOccupancy = batchOccupancy.get(warehouseCode);
    if (!warehouseOccupancy) return startDate;

    for (let i = 0; i < maxDays; i++) {
      const date = dateTimeUtils.addDays(startDate, i);
      const dateStr = dateTimeUtils.formatDate(date);

      // 跳过周末
      if (this.isWeekend(date)) continue;

      const occupancy = warehouseOccupancy.get(dateStr);
      if (!occupancy || occupancy.remainingCapacity > 0) {
        return date;
      }
    }

    return null; // 无可用日期
  }
}
```

**预期效果**：1500 次查询 → 1 次查询。

#### 1.3 并发处理优化

**问题**：货柜串行处理，未利用多核 CPU。

**方案**：使用 Promise.all 并行处理。

```typescript
// backend/src/services/IntelligentSchedulingService.ts

export class IntelligentSchedulingService {
  private readonly BATCH_SIZE = 10; // 每批并发数

  /**
   * 批量排产（并发优化版）
   */
  async batchScheduleOptimized(request: ScheduleRequest): Promise<BatchScheduleResponse> {
    const containers = await this.getContainersToSchedule(request);

    // 分批处理
    const results: ScheduleResult[] = [];
    for (let i = 0; i < containers.length; i += this.BATCH_SIZE) {
      const batch = containers.slice(i, i + this.BATCH_SIZE);

      // 并发处理当前批次
      const batchResults = await Promise.all(
        batch.map(container => this.scheduleSingleOptimized(container, request))
      );

      results.push(...batchResults);

      // 批次间隔（避免数据库连接池耗尽）
      if (i + this.BATCH_SIZE < containers.length) {
        await this.delay(50);
      }
    }

    return {
      success: true,
      total: containers.length,
      successCount: results.filter(r => r.success).length,
      failedCount: results.filter(r => !r.success).length,
      results
    };
  }

  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  private async scheduleSingleOptimized(
    container: Container,
    request: ScheduleRequest
  ): Promise<ScheduleResult> {
    // 复用批量档期（由调用方预加载）
    const batchOccupancy = await this.occupancyCalculator.getBatchWarehouseOccupancy(
      this.getCandidateWarehouseCodes(),
      new Date(),
      30
    );

    return this.scheduleSingleWithBatch(container, request, batchOccupancy);
  }
}
```

**预期效果**：100 柜处理时间从 45 秒降至 5-10 秒。

### 二、中期优化（1-2 月）

#### 2.1 规则引擎

**问题**：策略选择逻辑硬编码，难以动态调整。

**方案**：引入轻量级规则引擎。

```typescript
// backend/src/engine/SchedulingRulesEngine.ts

interface RuleContext {
  container: Container;
  warehouse: Warehouse;
  truckingCompany: TruckingCompany;
  lastFreeDate: Date;
  remainingDays: number;
  currentDate: Date;
}

interface Rule {
  name: string;
  priority: number; // 优先级（数字越小越先执行）
  condition: (ctx: RuleContext) => boolean;
  action: (ctx: RuleContext) => Partial<UnloadOption>;
}

// 规则定义
const schedulingRules: Rule[] = [
  {
    name: '免费期内优先Direct',
    priority: 1,
    condition: (ctx) => ctx.remainingDays > 3 && !ctx.truckingCompany.hasYard,
    action: (ctx) => ({ strategy: 'Direct' })
  },
  {
    name: '超期但有堆场考虑DropOff',
    priority: 2,
    condition: (ctx) => ctx.remainingDays <= 0 && ctx.truckingCompany.hasYard,
    action: (ctx) => ({ strategy: 'Drop off' })
  },
  {
    name: '临近超期加急处理',
    priority: 3,
    condition: (ctx) => ctx.remainingDays > 0 && ctx.remainingDays <= 3,
    action: (ctx) => ({ strategy: 'Expedited' })
  }
];

export class SchedulingRulesEngine {
  /**
   * 执行规则引擎
   */
  evaluate(context: RuleContext): Partial<UnloadOption> {
    // 按优先级排序
    const sortedRules = [...schedulingRules].sort((a, b) => a.priority - b.priority);

    for (const rule of sortedRules) {
      if (rule.condition(context)) {
        return rule.action(context);
      }
    }

    // 默认策略
    return { strategy: 'Direct' };
  }

  /**
   * 动态添加规则（运行时）
   */
  addRule(rule: Rule): void {
    schedulingRules.push(rule);
  }

  /**
   * 获取所有规则
   */
  getRules(): Rule[] {
    return [...schedulingRules];
  }
}
```

**预期效果**：策略调整无需修改代码，配置化。

#### 2.2 混合优化算法

**问题**：启发式搜索精度不足，模拟退火速度慢。

**方案**：启发式搜索 + 模拟退火微调。

```typescript
// backend/src/services/HybridOptimizer.ts

export class HybridOptimizer {
  private readonly HEURISTIC_WINDOW = 7;   // 启发式搜索窗口（天）
  private readonly SA_ITERATIONS = 1000;   // 模拟退火迭代次数
  private readonly SA_TEMPERATURE = 1000;  // 初始温度
  private readonly SA_COOLING = 0.995;      // 冷却速率

  /**
   * 两阶段优化
   */
  async optimize(input: UnloadInput): Promise<OptimalUnloadResult> {
    // 阶段1：启发式搜索（快速找到满意解）
    const heuristicResult = await this.heuristicSearch(input);

    // 阶段2：模拟退火微调（在满意解附近搜索更优解）
    const optimizedResult = await this.simulatedAnnealing(
      heuristicResult,
      input
    );

    return optimizedResult;
  }

  /**
   * 阶段1：启发式搜索
   */
  private async heuristicSearch(input: UnloadInput): Promise<UnloadOption> {
    // 现有 suggestOptimalUnloadDate 逻辑
    const candidates = await this.generateCandidates(input, this.HEURISTIC_WINDOW);
    return this.selectBest(candidates);
  }

  /**
   * 阶段2：模拟退火微调
   */
  private async simulatedAnnealing(
    initial: UnloadOption,
    input: UnloadInput
  ): Promise<OptimalUnloadResult> {
    let current = { ...initial };
    let best = current;
    let temperature = this.SA_TEMPERATURE;

    for (let i = 0; i < this.SA_ITERATIONS; i++) {
      // 生成邻域解（日期±1-2天）
      const neighbor = this.generateNeighbor(current, input);

      // 评估成本
      const currentCost = await this.evaluateCost(current, input);
      const neighborCost = await this.evaluateCost(neighbor, input);

      // 接受准则
      const delta = neighborCost - currentCost;
      if (delta < 0 || Math.random() < Math.exp(delta / temperature)) {
        current = neighbor;
      }

      // 更新最优解
      if (neighborCost < await this.evaluateCost(best, input)) {
        best = neighbor;
      }

      // 降温
      temperature *= this.SA_COOLING;

      // 早停：温度降至阈值
      if (temperature < 1) break;
    }

    return {
      ...best,
      costBreakdown: await this.getCostBreakdown(best, input)
    };
  }

  /**
   * 生成邻域解
   */
  private generateNeighbor(
    current: UnloadOption,
    input: UnloadInput
  ): UnloadOption {
    const offset = Math.floor(Math.random() * 3) - 1; // -1, 0, 1
    return {
      ...current,
      plannedPickupDate: dateTimeUtils.addDays(current.plannedPickupDate, offset)
    };
  }
}
```

**预期效果**：精度接近全局最优，速度保持毫秒级。

#### 2.3 多仓库/车队智能选择

**问题**：当前只考虑单一仓库/车队，未全局最优。

**方案**：建立候选方案池，全局最优搜索。

```typescript
// backend/src/services/WarehouseTruckingOptimizer.ts

export interface CandidateSolution {
  warehouse: Warehouse;
  truckingCompany: TruckingCompany;
  plannedPickupDate: Date;
  strategy: 'Direct' | 'Drop off' | 'Expedited';
  totalCost: number;
  score: number; // 综合评分
}

/**
 * 多仓库车队联合优化
 */
export class WarehouseTruckingOptimizer {
  /**
   * 生成所有可行方案
   */
  async generateAllCandidates(
    container: Container,
    warehouses: Warehouse[],
    truckingCompanies: TruckingCompany[]
  ): Promise<CandidateSolution[]> {
    const candidates: CandidateSolution[] = [];

    for (const warehouse of warehouses) {
      for (const trucking of truckingCompanies) {
        for (const strategy of ['Direct', 'Drop off', 'Expedited'] as const) {
          // 评估不同日期（±7天）
          for (let offset = -7; offset <= 7; offset++) {
            const pickupDate = dateTimeUtils.addDays(new Date(), offset);

            if (this.isValid(warehouse, trucking, pickupDate, strategy)) {
              const cost = await this.calculateTotalCost(
                container, warehouse, trucking, pickupDate, strategy
              );

              candidates.push({
                warehouse,
                truckingCompany: trucking,
                plannedPickupDate: pickupDate,
                strategy,
                totalCost: cost,
                score: this.calculateScore(cost, warehouse, trucking, offset)
              });
            }
          }
        }
      }
    }

    // 按综合评分排序
    return candidates.sort((a, b) => b.score - a.score);
  }

  /**
   * 计算综合评分
   *
   * 权重配置：
   * - 成本权重：0.6
   * - 时间权重：0.3（越接近最佳日期越好）
   * - 仓库权重：0.1（优先特定仓库）
   */
  private calculateScore(
    cost: number,
    warehouse: Warehouse,
    trucking: TruckingCompany,
    dateOffset: number
  ): number {
    const costWeight = 0.6;
    const timeWeight = 0.3;
    const warehouseWeight = 0.1;

    // 成本归一化（假设最大成本 5000）
    const costScore = (5000 - cost) / 5000;

    // 时间评分（offset=0 最优）
    const timeScore = 1 - Math.abs(dateOffset) / 7;

    // 仓库评分（可配置）
    const warehouseScore = warehouse.priority || 0.5;

    return (
      costScore * costWeight +
      timeScore * timeWeight +
      warehouseScore * warehouseWeight
    );
  }

  /**
   * 选择最优方案
   */
  selectBest(candidates: CandidateSolution[]): CandidateSolution {
    if (candidates.length === 0) {
      throw new Error('No feasible solution found');
    }

    return candidates[0]; // 已按评分排序
  }
}
```

// backend/src/events/SchedulingEventBus.ts

export enum SchedulingEventType {
  CONTAINER_ARRIVED = 'container:arrived',
  CONTAINER_DEPARTED = 'container:departed',
  WAREHOUSE_CAPACITY_CHANGED = 'warehouse:capacity_changed',
  TRUCKING_AVAILABILITY_CHANGED = 'trucking:availability_changed',
  DEMURRAGE_STANDARD_UPDATED = 'demurrage:standard_updated'
}

export interface SchedulingEvent {
  type: SchedulingEventType;
  payload: any;
  timestamp: Date;
}

/**
 * 事件总线
 */
export class SchedulingEventBus {
  private handlers = new Map<SchedulingEventType, Function[]>();

  /**
   * 订阅事件
   */
  subscribe(eventType: SchedulingEventType, handler: Function): void {
    if (!this.handlers.has(eventType)) {
      this.handlers.set(eventType, []);
    }
    this.handlers.get(eventType)!.push(handler);
  }

  /**
   * 发布事件
   */
  async publish(event: SchedulingEvent): Promise<void> {
    const handlers = this.handlers.get(event.type) || [];

    await Promise.all(
      handlers.map(handler => handler(event))
    );
  }
}

// 使用示例
export class PortOperationService {
  constructor(private eventBus: SchedulingEventBus) {}

  async processArrival(containerNumber: string): Promise<void> {
    // 处理到港逻辑...

    // 发布事件触发排产
    await this.eventBus.publish({
      type: SchedulingEventType.CONTAINER_ARRIVED,
      payload: { containerNumber },
      timestamp: new Date()
    });
  }
}

#### 3.2 多目标优化

**问题**：只优化成本，未考虑其他目标。

**方案**：NSGA-II 多目标遗传算法。

```typescript
// backend/src/services/MultiObjectiveOptimizer.ts

interface OptimizationObjectives {
  minimizeCost: number;        // 最小化总成本
  minimizeOverdue: number;     // 最小化超期货柜数
  maximizeUtilization: number; // 最大化仓库利用率
  maximizeCustomerSatisfaction: number; // 最大化客户满意度
}

export class MultiObjectiveOptimizer {
  private readonly POPULATION_SIZE = 100;
  private readonly GENERATIONS = 50;
  private readonly CROSSOVER_RATE = 0.9;
  private readonly MUTATION_RATE = 0.1;

  /**
   * NSGA-II 多目标优化
   */
  async optimize(
    containers: Container[],
    objectives: OptimizationObjectives
  ): Promise<SchedulingResult[]> {
    // 初始化种群
    let population = this.initializePopulation(containers);

    // 迭代进化
    for (let gen = 0; gen < this.GENERATIONS; gen++) {
      // 评估适应度
      const fitness = this.evaluateFitness(population, objectives);

      // 非支配排序
      const fronts = this.fastNonDominatedSort(fitness);

      // 计算拥挤度距离
      const crowdingDistance = this.calculateCrowdingDistance(fronts, fitness);

      // 选择、交叉、变异
      population = this.evolve(population, fronts, crowdingDistance);
    }

    // 返回 Pareto 最优前沿
    return this.getParetoFront(population);
  }

  /**
   * 快速非支配排序
   */
  private fastNonDominatedSort(fitness: OptimizationObjectives[]): number[][] {
    const n = fitness.length;
    const dominationCount = new Array(n).fill(0);
    const dominatedSet: number[][] = Array.from({ length: n }, () => []);
    const fronts: number[][] = [[]];

    for (let i = 0; i < n; i++) {
      for (let j = 0; j < n; j++) {
        if (i === j) continue;

        if (this.dominates(fitness[i], fitness[j])) {
          dominatedSet[i].push(j);
        } else if (this.dominates(fitness[j], fitness[i])) {
          dominationCount[i]++;
        }
      }

      if (dominationCount[i] === 0) {
        fronts[0].push(i);
      }
    }

    // 构建前沿
    let frontIndex = 0;
    while (fronts[frontIndex].length > 0) {
      const nextFront: number[] = [];
      for (const i of fronts[frontIndex]) {
        for (const j of dominatedSet[i]) {
          dominationCount[j]--;
          if (dominationCount[j] === 0) {
            nextFront.push(j);
          }
        }
      }
      frontIndex++;
      if (nextFront.length > 0) {
        fronts.push(nextFront);
      }
    }

    return fronts;
  }

  /**
   * 判断解 A 是否支配解 B
   */
  private dominates(a: OptimizationObjectives, b: OptimizationObjectives): boolean {
    return (
      a.minimizeCost <= b.minimizeCost &&
      a.minimizeOverdue <= b.minimizeOverdue &&
      a.maximizeUtilization >= b.maximizeUtilization &&
      (a.minimizeCost < b.minimizeCost ||
        a.minimizeOverdue < b.minimizeOverdue ||
        a.maximizeUtilization > b.maximizeUtilization)
    );
  }
}
```
