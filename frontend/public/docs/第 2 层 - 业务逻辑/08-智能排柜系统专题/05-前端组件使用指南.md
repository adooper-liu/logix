## 一、SchedulingVisual 组件

### 1.1 基本用法

**路径**: `@/views/scheduling/SchedulingVisual.vue`

**用途**: 智能排产主页面，提供完整的排产流程 UI

```vue
<template>
  <SchedulingVisual
    :country="countryCode"
    :initial-date-range="[startDate, endDate]"
    :containers="containerNumbers"
    @complete="handleComplete"
    @error="handleError"
  />
</template>

<script setup lang="ts">
import SchedulingVisual from '@/views/scheduling/SchedulingVisual.vue'

const countryCode = 'US'
const startDate = new Date('2026-03-20')
const endDate = new Date('2026-04-20')
const containerNumbers = 'HMMU6232153,HMMU6232154'

const handleComplete = (result: any) => {
  console.log('排产完成:', result)
}

const handleError = (error: any) => {
  console.error('排产失败:', error)
}
</script>
```

---

### 1.2 Props 详解

```typescript
interface SchedulingVisualProps {
  /**
   * 国家代码（可选）
   * 用于过滤该国家的货柜
   */
  country?: string

  /**
   * 初始日期范围（可选）
   * 默认值：[今天，今天 +30 天]
   */
  initialDateRange?: [Date, Date]

  /**
   * 柜号列表（可选）
   * 逗号分隔的字符串，如 "HMMU6232153,HMMU6232154"
   * 如果不传，则查询该国家下所有待排产的货柜
   */
  containers?: string
}
```

---

### 1.3 Events 详解

```typescript
interface SchedulingVisualEvents {
  /**
   * 排产完成事件
   * payload: 排产结果对象
   */
  (e: 'complete', result: {
    scheduledContainers: Array<{
      containerNumber: string
      warehouseCode: string
      truckingCompanyId: string
      plannedPickupDate: string
      plannedUnloadDate: string
      plannedReturnDate: string
      unloadMode: 'Live load' | 'Drop off'
      status: 'scheduled' | 'failed'
    }>
    summary: {
      totalContainers: number
      scheduledCount: number
      failedCount: number
      estimatedTotalCost: number
    }
  }): void

  /**
   * 错误事件
   * payload: 错误对象
   */
  (e: 'error', error: Error): void
}
```

### 1.4 完整示例

```vue
<template>
  <div class="scheduling-page">
    <!-- 面包屑导航 -->
    <el-breadcrumb separator="/">
      <el-breadcrumb-item>首页</el-breadcrumb-item>
      <el-breadcrumb-item>智能排产</el-breadcrumb-item>
    </el-breadcrumb>

    <!-- 智能排产主组件 -->
    <SchedulingVisual
      ref="schedulingRef"
      :country="selectedCountry"
      :initial-date-range="dateRange"
      :containers="selectedContainers"
      @complete="onScheduleComplete"
      @error="onScheduleError"
    />

    <!-- 排产结果导出对话框 -->
    <ExportDialog
      v-model:visible="exportDialogVisible"
      :data="exportData"
    />
  </div>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import SchedulingVisual from '@/views/scheduling/SchedulingVisual.vue'
import ExportDialog from './components/ExportDialog.vue'

// 状态
const schedulingRef = ref<InstanceType<typeof SchedulingVisual>>()
const exportDialogVisible = ref(false)
const exportData = ref<any>(null)

// 配置
const selectedCountry = 'US'
const dateRange: [Date, Date] = [
  new Date(),
  new Date(new Date().setDate(new Date().getDate() + 30))
]
const selectedContainers = 'HMMU6232153,HMMU6232154,HMMU6232155'

// 事件处理
const onScheduleComplete = (result: any) => {
  console.log('排产完成:', result)
  
  // 显示成功提示
  ElMessage.success(`排产完成！成功 ${result.summary.scheduledCount} 个，失败 ${result.summary.failedCount} 个`)
  
  // 保存导出数据
  exportData.value = result
  
  // 询问是否导出
  ElMessageBox.confirm(
    '排产已完成，是否导出结果？',
    '提示',
    {
      confirmButtonText: '导出',
      cancelButtonText: '取消',
      type: 'success'
    }
  ).then(() => {
    exportDialogVisible.value = true
  }).catch(() => {
    // 用户取消
  })
}

const onScheduleError = (error: any) => {
  console.error('排产失败:', error)
  
  // 显示错误提示
  ElMessage.error({
    message: error.response?.data?.message || '排产失败，请稍后重试',
    duration: 5000
  })
}
</script>

<style scoped lang="scss">
.scheduling-page {
  padding: 24px;
  background: #f5f7fa;
  min-height: 100vh;
}
</style>
```

### 4.1 完整排产流程页面

```vue
<template>

    <!-- 顶部工具栏 -->

      <el-button type="primary" @click="executeBatchSchedule">

        <el-icon><Calendar /></el-icon> {{ showGantt ? '隐藏' : '显示' }}甘特图
      </el-button>
    </div>

    <!-- 左侧：排产表单 -->
    <div class="main-content" :class="{ 'show-gantt': showGantt }">
      <SchedulingVisual

        :country="countryCode"
        :initial-date-range="dateRange"
        @complete="handleScheduleComplete"
      />
    </div>

    <!-- 右侧：甘特图预览 -->
    <div v-if="showGantt" class="gantt-panel">
      <ContainerGanttChart
        :containers="ganttData"
        :range-type="30"
        :editable="false"
      />
    </div>
  </div>

<script setup lang="ts">
import { ref, computed } from 'vue'
import SchedulingVisual from '@/views/scheduling/SchedulingVisual.vue'
import ContainerGanttChart from '@/components/common/ContainerGanttChart.vue'
import { Cpu, Calendar } from '@element-plus/icons-vue'
import api from '@/services/api'

const schedulingRef = ref<InstanceType<typeof SchedulingVisual>>()
const showGantt = ref(false)
const countryCode = 'US'

  new Date(),
  new Date(new Date().setDate(new Date().getDate() + 30))
]

  try {
    const response = await api.post('/scheduling/batch', {
      containerNumbers: [],  // 空表示查询所有

      endDate: dateRange[1].toISOString().split('T')[0],

  } catch (error) {
    ElMessage.error('排产失败：' + (error as Error).message)
  }
}

const handleScheduleComplete = (result: any) => {
  console.log('排产完成:', result)
  // 可以在这里更新甘特图或其他 UI
}
</script>
```

## 五、最佳实践

### 5.1 性能优化

```vue
<!-- ❌ 坏的做法：在模板中直接调用方法 -->
<template>
  <div>
    <div v-for="container in getFilteredContainers()" :key="container.id">
      {{ container.name }}
    </div>
  </div>
</template>

<script setup>
const getFilteredContainers = () => {
  // 复杂计算...
  return containers.filter(...)
}
</script>

<!-- ✅ 好的做法：使用计算属性 -->
<template>
  <div>
    <div v-for="container in filteredContainers" :key="container.id">
      {{ container.name }}
    </div>
  </div>
</template>

<script setup>
import { computed } from 'vue'

const filteredContainers = computed(() => {
  return containers.filter(c => c.status === 'pending')
})
</script>
```

### 5.2 错误处理

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { ElMessage } from 'element-plus'

const loading = ref(false)
const error = ref<string | null>(null)

const executeSchedule = async () => {
  loading.value = true
  error.value = null

  try {
    const response = await api.post('/scheduling/batch', { ... })
    ElMessage.success('排产成功')
    return response.data
  } catch (err) {
    const errorMessage = (err as any).response?.data?.message || '未知错误'
    error.value = errorMessage
    ElMessage.error(errorMessage)
    throw err
  } finally {
    loading.value = false
  }
}
</script>

<template>
  <div>
    <el-button 
      :loading="loading" 
      @click="executeSchedule"
    >
      执行排产
    </el-button>
    
    <el-alert
      v-if="error"
      type="error"
      :title="error"
      show-icon
      closable
    />
  </div>
</template>
```
