### 1.2 Props 详解

```typescript
interface SimpleGanttChartProps {
  /**
   * 初始显示范围类型

}
```

### 3.2 自定义扩展

```typescript
// 创建自定义 composable
import { useGanttLogic } from '@/components/common/gantt/useGanttLogic'
import { computed } from 'vue'

export function useCustomGantt() {
  // 复用基础逻辑
  const baseGantt = useGanttLogic()

  // 添加自定义状态
  const selectedContainers = ref<string[]>([])
  const showOnlyAlerts = ref(false)

  // 添加自定义计算属性
  const filteredContainers = computed(() => {
    let result = baseGantt.containers.value

    // 只显示预警货柜
    if (showOnlyAlerts.value) {
      result = result.filter(c => baseGantt.hasAlert(c))
    }

    // 选中的货柜
    if (selectedContainers.value.length > 0) {
      result = result.filter(c => selectedContainers.value.includes(c.containerNumber))
    }

    return result
  })

  // 添加自定义方法
  async function toggleSelection(containerNumber: string) {
    const index = selectedContainers.value.indexOf(containerNumber)

    if (index >= 0) {
      selectedContainers.value.splice(index, 1)
    } else {
      selectedContainers.value.push(containerNumber)
    }
  }

  function toggleAlertsFilter() {
    showOnlyAlerts.value = !showOnlyAlerts.value
  }

  return {
    ...baseGantt,
    selectedContainers,
    showOnlyAlerts,
    filteredContainers,
    toggleSelection,
    toggleAlertsFilter,
  }
}
```

### 5.1 性能优化

```vue
<script setup lang="ts">
import { computed } from 'vue'

// ✅ 好的做法：使用计算属性缓存
const finalFilteredContainers = computed(() => {
  return containers.value.filter(container => {
    // 复杂的多条件过滤
    const matchSearch = !keyword.value ||
      container.containerNumber.toLowerCase().includes(keyword.value.toLowerCase())

    const matchPort = !selectedPort.value ||
      container.destinationPort === selectedPort.value

    const matchStatus = !selectedStatus.value ||
      container.logisticsStatus === selectedStatus.value

    const matchAlerts = !showOnlyAlerts.value ||
      hasAlert(container)

    return matchSearch && matchPort && matchStatus && matchAlerts
  })
})
```

### 5.2 错误处理

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { ElMessage } from 'element-plus'

const error = ref<string | null>(null)
const loading = ref(false)

async function loadData() {
  loading.value = true
  error.value = null

  try {
    const response = await api.get('/containers/gantt', { params })
    containers.value = response.data.data.containers
  } catch (err) {
    error.value = '数据加载失败，请稍后重试'
    ElMessage.error({
      message: error.value,
      duration: 5000,
    })
  } finally {
    loading.value = false
  }
}
</script>

<template>
  <div>
    <el-alert v-if="error" type="error" :title="error" show-icon closable />

    <div v-loading="loading">
      <!-- 甘特图内容 -->
    </div>
  </div>
</template>
```
