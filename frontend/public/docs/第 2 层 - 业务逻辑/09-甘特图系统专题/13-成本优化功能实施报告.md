### 3.3 useGanttLogic 集成

**新增状态**:

```typescript
// 🎯 成本优化面板状态
const showOptimizationPanel = ref(false)
const optimizationResult = ref<any>(null)
```

**触发逻辑**:
```typescript
const triggerCostOptimization = async (
  container: Container,
  updateData: Record<string, string>
) => {
  // 1. 获取必要参数
  const trucking = container.truckingTransports?.[0]
  const warehouse = container.warehouseOperations?.[0]
  
  // 2. 构建请求参数
  const basePickupDate = updateData.plannedPickupDate || trucking.plannedPickupDate
  
  // 3. 调用优化 API
  const result = await costOptimizationService.optimizeContainer(
    container.containerNumber,
    {
      warehouseCode: warehouse.warehouseId,
      truckingCompanyId: trucking.truckingCompanyId,
      basePickupDate,
    }
  )
  
  // 4. 显示优化面板（仅当有节省时）
  if (result.success && result.data.savings > 0) {
    optimizationResult.value = { ...result.data }
    showOptimizationPanel.value = true
  }
}
```

**调用时机**:

```typescript
import { debounce } from 'lodash-es'

const debouncedOptimization = debounce(async (container, updateData) => {
  await triggerCostOptimization(container, updateData)
}, 500)

// 调用时使用
debouncedOptimization(container, updateData)
```

#### 2. 结果缓存

**问题**: 相同参数的重复调用浪费资源

**解决方案**:
```typescript
const optimizationCache = new Map<string, {
  data: OptimizeContainerResponse
  timestamp: number
}>()

const CACHE_TTL = 5 * 60 * 1000 // 5分钟

function getCachedResult(key: string): OptimizeContainerResponse | null {
  const cached = optimizationCache.get(key)
  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.data
  }
  return null
}

function setCachedResult(key: string, data: OptimizeContainerResponse) {
  optimizationCache.set(key, { data, timestamp: Date.now() })
}
```
