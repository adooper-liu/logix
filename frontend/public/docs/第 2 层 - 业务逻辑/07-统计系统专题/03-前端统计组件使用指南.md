## 📋 概述

本文档详细介绍 LogiX 统计模块的前端组件使用，包括筛选条件映射、统计卡片、甘特图统计面板等核心组件。

### 核心组件

1. **统计卡片组件** - 展示各维度统计数据
2. **GanttStatisticsPanel** - 甘特图统计面板
3. **FilterConditions** - 筛选条件常量
4. **useContainerCountdown** - 倒计时 Composable

### 1.2 使用示例

```vue
<script setup lang="ts">
import { CONDITION_TO_SERVICE_MAP } from '@/constants/FilterConditions';

// 根据筛选条件调用对应的 API
async function loadContainers(filterCondition: string) {
  const serviceType = CONDITION_TO_SERVICE_MAP[filterCondition];
```

## 四、倒计时 Composable

### 4.1 useContainerCountdown

**文件**: `frontend/src/composables/useContainerCountdown.ts`

**用途**: 计算货柜距离免费期截止的剩余天数

**使用示例**:

```vue
<template>
  <div class="container-countdown">
    <span v-if="countdown.daysRemaining !== null">
      剩余 {{ countdown.daysRemaining }} 天
    </span>
    <span v-else>无免费期</span>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import { useContainerCountdown } from '@/composables/useContainerCountdown';

const props = defineProps<{
  lastFreeDate?: string; // 最晚提柜日
}>();

const { countdown } = useContainerCountdown(() => props.lastFreeDate);
</script>
```

---

### 4.2 composable 实现

```typescript
import { computed } from 'vue';
import dayjs from 'dayjs';

export function useContainerCountdown(lastFreeDateGetter: () => string | undefined) {
  const countdown = computed(() => {
    const lastFreeDate = lastFreeDateGetter();
    
    if (!lastFreeDate) {
      return {
        daysRemaining: null,
        isExpired: false,
        isUrgent: false,
        isWarning: false,
        status: 'noLastFreeDate'
      };
    }
    
    const today = dayjs().startOf('day');
    const deadline = dayjs(lastFreeDate).startOf('day');
    const daysRemaining = deadline.diff(today, 'day');
    
    let status: 'expired' | 'urgent' | 'warning' | 'normal';
    
    if (daysRemaining < 0) {
      status = 'expired';
    } else if (daysRemaining <= 3) {
      status = 'urgent';
    } else if (daysRemaining <= 7) {
      status = 'warning';
    } else {
      status = 'normal';
    }
    
    return {
      daysRemaining,
      isExpired: daysRemaining < 0,
      isUrgent: daysRemaining >= 0 && daysRemaining <= 3,
      isWarning: daysRemaining > 3 && daysRemaining <= 7,
      isNormal: daysRemaining > 7,
      status
    };
  });
  
  return {
    countdown
  };
}
```

## 五、组合使用示例

### 5.1 完整 Shipments 页面

```vue
<template>
  <div class="shipments-page">
    <!-- 统计卡片区 -->
    <StatisticsCards 
      :statistics="statistics"
      @select="handleStatSelect"
    />
    
    <!-- 筛选工具栏 -->
    <div class="toolbar">
      <el-select v-model="currentFilter" placeholder="选择筛选维度">
        <el-option label="按到港" value="arrival" />
        <el-option label="按 ETA" value="eta" />
        <el-option label="按计划提柜" value="plannedPickup" />
        <el-option label="按最晚提柜" value="lastPickup" />
        <el-option label="按最晚还箱" value="lastReturn" />
      </el-select>
      
      <el-button @click="loadContainers">查询</el-button>
    </div>
    
    <!-- 货柜列表 -->
    <ContainerList 
      :containers="containers"
      :loading="containersLoading"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue';
import StatisticsCards from './components/StatisticsCards.vue';
import ContainerList from './components/ContainerList.vue';
import ContainerService from '@/services/container';

const statistics = ref({
  arrivalToday: 0,
  arrivedBeforeTodayNotPickedUp: 0,
  // ... 其他统计字段
});

const currentFilter = ref('arrival');
const containers = ref([]);
const containersLoading = ref(false);

// 加载统计数据
async function loadStatistics() {
  const data = await ContainerService.getStatistics();
  statistics.value = data;
}

// 加载货柜列表
async function loadContainers() {
  containersLoading.value = true;
  try {
    const response = await ContainerService.getContainers({
      filterCondition: currentFilter.value
    });
    containers.value = response.items;
  } finally {
    containersLoading.value = false;
  }
}

// 处理统计卡片点击
function handleStatSelect(filterCondition: string) {
  currentFilter.value = CONDITION_TO_SERVICE_MAP[filterCondition];
  loadContainers();
}

onMounted(() => {
  loadStatistics();
  loadContainers();
});
</script>
```

## 六、最佳实践

### 6.1 性能优化

```typescript
// ✅ 好的做法：缓存统计数据
const statisticsCache = new Map<string, any>();

async function getStatistics(key: string) {
  // 检查缓存（5 分钟内有效）
  const cached = statisticsCache.get(key);
  if (cached && Date.now() - cached.timestamp < 5 * 60 * 1000) {
    return cached.data;
  }
  
  // 从 API 获取
  const data = await ContainerService.getStatistics();
  
  // 写入缓存
  statisticsCache.set(key, {
    data,
    timestamp: Date.now()
  });
  
  return data;
}
```

### 6.2 错误处理

```typescript
async function loadStatistics() {
  try {
    const data = await ContainerService.getStatistics();
    statistics.value = data;
  } catch (error) {
    console.error('[Statistics] Failed to load:', error);
    
    // 显示错误提示
    ElMessage.error('加载统计数据失败');
    
    // 设置默认值（全 0）
    statistics.value = getDefaultStatistics();
  }
}

function getDefaultStatistics() {
  return {
    arrivalToday: 0,
    arrivedBeforeTodayNotPickedUp: 0,
    // ... 所有字段初始化为 0
  };
}
```
