## 📊 一、快速概览

### 核心目标

```
从"能排产"升级到"排得好"

当前 (能排产):
输入 → 找到第一个可用仓库 → 输出 (可能产生高额滞港费)

目标 (排得好):
输入 → 生成所有可行方案 → 评估每个方案成本 → 选择最优方案 → 输出 + 成本明细
```

## 🎯 二、核心改进点

### 改进 1: 成本驱动的决策机制

```typescript
// ❌ 当前逻辑（可行性驱动）
const unloadDate = await findEarliestAvailableWarehouse(warehouses, pickupDate);

// ✅ 新逻辑（成本驱动）
const options = await generateAllFeasibleOptions(container, pickupDate);
for (const option of options) {
  option.totalCost = await evaluateTotalCost(option);
}
const best = options.sort((a, b) => a.totalCost - b.totalCost)[0];
const unloadDate = best.date;
```

---

### 改进 2: 多方案择优

```
生成三种方案:
1. Direct (直接卸柜) - 默认方案
2. Drop off (先卸堆场) - 当 Direct 会产生高额滞港费时
3. Expedited (协调加急) - 当免费期即将到期时

评估每个方案的总成本:
总成本 = 滞港费 + 滞箱费 + 堆存费 + 运输费 + 操作费

选择成本最低的方案 ✅
```

```typescript
// backend/src/services/schedulingCostOptimizer.service.ts
@Injectable()
export class SchedulingCostOptimizerService {
  constructor(
    @InjectRepository(DictSchedulingConfig)
    private schedulingConfigRepo: Repository<DictSchedulingConfig>,
    private demurrageService: DemurrageService,
    private warehouseService: WarehouseService,
    private truckingService: TruckingService
  ) {}
  
  async evaluateTotalCost(option: UnloadOption): Promise<CostBreakdown> {
    // 1. 滞港费（复用 demurrage.service）
    const demurrage = await this.demurrageService.predictDemurrageForUnloadDate(
      option.containerNumber,
      option.unloadDate
    );
    
    // 2. 滞箱费（可选链调用，Phase 2 可暂不计入）
    const detention = this.demurrageService.predictDetentionForReturnDate
      ? await this.demurrageService.predictDetentionForReturnDate(
          option.containerNumber,
          option.returnDate || option.unloadDate
        )
      : { detentionCost: 0 };
    
    // 3. 堆存费（Drop off 模式，从配置读取）
    if (option.strategy === 'Drop off') {
      const storageDays = this.calculateStorageDays(option);
      const rate = await this.getConfigNumber('external_storage_daily_rate', 50);
      breakdown.storageCost = rate * storageDays;
    }
    
    // 4. 运输费（Phase 2 暂置 0）
    breakdown.transportationCost = 0;
    
    // 5. 操作费（加急费，从配置读取）
    if (option.strategy === 'Expedited') {
      breakdown.handlingCost = await this.getConfigNumber('expedited_handling_fee', 50);
    }
    
    // 6. 总成本
    breakdown.totalCost = 
      breakdown.demurrageCost +
      breakdown.detentionCost +
      breakdown.storageCost +
      breakdown.transportationCost +
      breakdown.handlingCost;
    
    return breakdown;
  }
}
```

```typescript
async predictDemurrageForUnloadDate(
  containerNumber: string,
  proposedUnloadDate: Date
): Promise<{
  lastFreeDate: Date;
  proposedUnloadDate: Date;
  demurrageDays: number;
  demurrageCost: number;
  tierBreakdown: Array<{ fromDay: number; toDay: number; days: number; ratePerDay: number; subtotal: number }>;
  currency: string;

  // 步骤:

  // 4. 计算计费天数（proposedUnloadDate - lastFreeDate）
  // 5. 调用 calculateSingleDemurrage() 计算费用
  
  // 注意:
  // - 支持四种免费天数基准模式
```

### 接口 3: generateAllFeasibleOptions

**位置**: `backend/src/services/schedulingCostOptimizer.service.ts`

```typescript
async generateAllFeasibleOptions(
  container: Container,
  pickupDate: Date,
  lastFreeDate: Date,
  searchWindowDays: number = 7
): Promise<UnloadOption[]> {
  const options: UnloadOption[] = [];
  
  // 1. 获取候选仓库列表（按国家 + 港口映射）
  const warehouses = await this.getCandidateWarehouses(
    container.countryCode,
    container.portCode
  );
  
  // 2. 为每个仓库生成 Direct 方案
  for (const warehouse of warehouses) {
    for (let offset = 0; offset < searchWindowDays; offset++) {
      const candidateDate = addDays(pickupDate, offset);
      
      // 跳过周末（如果配置了）
      if (this.isWeekend(candidateDate) && this.shouldSkipWeekends()) {
        continue;
      }
      
      // 检查仓库档期
      if (!await this.isWarehouseAvailable(warehouse, candidateDate)) {
        continue;
      }
      
      options.push({
        containerNumber: container.containerNumber,
        warehouse,
        unloadDate: candidateDate,
        strategy: 'Direct',
        isWithinFreePeriod: candidateDate <= lastFreeDate
      });
    }
  }
  
  // 3. 生成 Drop off 方案（车队须来自 warehouse_trucking_mapping）
  const dropOffOptions = await this.generateDropOffOptions(
    container,
    pickupDate,
    lastFreeDate
  );
  options.push(...dropOffOptions);
  
  // 4. 生成 Expedited 方案（协调加急）
  const expeditedOptions = await this.generateExpeditedOptions(
    container,
    lastFreeDate
  );
  options.push(...expeditedOptions);
  
  return options;
}
```
