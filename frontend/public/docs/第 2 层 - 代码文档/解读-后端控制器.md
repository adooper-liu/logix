```
                             │

```

## 八、控制器代码结构示例

```typescript
@Controller('containers')
@UseGuards(AuthGuard)
export class ContainerController {
  constructor(
    private readonly containerService: ContainerService,
    private readonly demurrageService: DemurrageService,
  ) {}

  @Get()
  @UsePipes(new ValidationPipe({ transform: true }))
  async getContainers(
    @Query() query: ContainerQueryDTO,
  ): Promise<ApiResponse<Container[]>> {
    const containers = await this.containerService.findAll(query);
    return { success: true, data: containers };
  }

  @Get(':containerNumber/demurrage')
  async getDemurrage(
    @Param('containerNumber') containerNumber: string,
  ): Promise<ApiResponse<DemurrageResult>> {
    const result = await this.demurrageService.calculateForContainer(
      containerNumber,
    );
    return { success: true, data: result };
  }
}
```

## 九、常用装饰器

| 装饰器 | 作用 |
|--------|------|
| @Controller(path) | 定义路由前缀 |
| @Get(path) | GET 请求 |
| @Post(path) | POST 请求 |
| @Put(path) | PUT 请求 |
| @Delete(path) | DELETE 请求 |
| @Param(name) | 路由参数 |
| @Query() | 查询参数 |
| @Body() | 请求体 |
| @UseGuards() | 权限守卫 |
| @UsePipes() | 参数管道 |
