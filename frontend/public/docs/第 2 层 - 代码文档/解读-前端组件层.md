```
  ├── Header (顶部导航)

  └── Main Content (主内容区)
        ├── RouterView
        │     ├── Dashboard
        │     │     └── StatsCard, Charts...
        │     ├── Shipments
        │     │     └── ContainerGanttChart / SimpleGanttChart
        │     ├── Scheduling
        │     │     └── CalendarCapacityView
        │     └── Demurrage

              ├── DateTimePicker / DateRangePicker
              ├── DurationDisplay / NodeDurationDisplay
              ├── AlertCenter / RiskCard
              └── MarkdownRenderer
```

## 六、组件通信模式

### 6.1 Props/Events 模式

```vue
<!-- 父组件 -->
<ChildComponent :prop="value" @event="handler" />

<!-- 子组件 -->
<script setup>
defineProps<{ prop: string }>()
defineEmits(['event'])
</script>
```

### 6.2 Pinia Store 模式

```typescript
// store
const useContainerStore = defineStore('container', () => { ... })

// 组件中使用
const containerStore = useContainerStore()
```

### 6.3 Provide/Inject 模式

```typescript
// 父组件
provide('theme', 'dark')

// 子组件
const theme = inject('theme')
```
