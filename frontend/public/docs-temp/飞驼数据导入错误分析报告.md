#### 1.1 日期时间格式错误

**错误表现**:
```
飞驼格式: 2025/7/18 1:37
预期格式: YYYY-MM-DD HH:mm:ss
解析结果: 可能转换为 2025-07-18 01:37:00 或解析失败
```

**代码位置**: `feituoImport.service.ts:105-116`

```typescript
function parseDate(val: unknown): Date | null {
  if (!val) return null;
  if (val instanceof Date) return val;
  const s = String(val).trim().replace(/\//g, '-');  // ✅ 已处理"/"替换为"-"
  const m = s.match(/^(\d{4})-?(\d{1,2})-?(\d{1,2})/);
  if (m) {
    const d = new Date(parseInt(m[1], 10), parseInt(m[2], 10) - 1, parseInt(m[3], 10));
    return isNaN(d.getTime()) ? null : d;  // ⚠️ 潜在问题：忽略时间部分
  }
  const d = new Date(s);
  return isNaN(d.getTime()) ? null : d;
}
```

**现有问题**:
- ⚠️ **时间部分丢失**: 正则表达式只匹配日期，忽略时间（如`1:37`）
- ⚠️ **时区问题**: 未处理时区转换，导致时间偏差
- ⚠️ **格式不严格**: 对`2025/7/18`和`2025/07/18`都接受，但`2025-7-18`可能失败

**实际影响**:
```typescript
// 输入: "2025/7/18 1:37"
// 步骤1: replace => "2025-7-18 1:37"
// 步骤2: match => ["2025-7-18", "2025", "7", "18"] (匹配成功，但忽略时间)
// 结果: new Date(2025, 6, 18) => 2025-07-18 00:00:00
// 丢失: 01:37 时间部分
```

**解决方案**:
```typescript
// 增强版parseDate函数
function parseDate(val: unknown): Date | null {
  if (!val) return null;
  if (val instanceof Date) return val;
  
  const s = String(val).trim().replace(/\//g, '-');
  
  // 支持完整日期时间格式: 2025/7/18 1:37 或 2025-7-18 01:37
  const fullMatch = s.match(/^(\d{4})[-/](\d{1,2})[-/](\d{1,2})\s+(\d{1,2}):(\d{1,2})/);
  if (fullMatch) {
    const [, year, month, day, hour, minute] = fullMatch;
    const d = new Date(parseInt(year), parseInt(month) - 1, parseInt(day), parseInt(hour), parseInt(minute));
    return isNaN(d.getTime()) ? null : d;
  }
  
  // 仅日期格式
  const dateMatch = s.match(/^(\d{4})[-/](\d{1,2})[-/](\d{1,2})/);
  if (dateMatch) {
    const [, year, month, day] = dateMatch;
    const d = new Date(parseInt(year), parseInt(month) - 1, parseInt(day));
    return isNaN(d.getTime()) ? null : d;
  }
  
  // 尝试标准Date解析
  const d = new Date(s);
  return isNaN(d.getTime()) ? null : d;
}
```

```typescript
// 增强版parseArrayField函数
function parseArrayField(val: unknown): string[] | null {
  if (!val) return null;
  if (Array.isArray(val)) return val.map(String);
  
  const str = String(val).trim();
  
  // JSON数组格式: ["a","b"]
  if (str.startsWith('[') && str.endsWith(']')) {
    try {
      return JSON.parse(str).map(String);
    } catch {
      // 解析失败，尝试手动分割
      return str.slice(1, -1).split(',').map(s => s.trim()).filter(Boolean);
    }
  }
  
  // 逗号分隔格式: a,b,c
  if (str.includes(',')) {
    return str.split(',').map(s => s.trim()).filter(Boolean);
  }
  
  // 单个值
  return [str];
}

// 安全的数字解析
function safeParseFloat(val: string | undefined): number | undefined {
  if (!val) return undefined;
  const num = parseFloat(val);
  return isNaN(num) ? undefined : num;
}
```

```typescript
// 方案1: 增加分组验证和警告
function validateFieldMapping(headers: string[], tableType: 1 | 2): string[] {
  const warnings: string[] = [];
  const occurrence: Record<string, number> = {};
  
  for (const h of headers) {
    occurrence[h] = (occurrence[h] ?? 0);
    const gid = getGroupForColumn(tableType, h, occurrence[h]);
    
    if (gid === 7) {
      warnings.push(`警告: 分组7(当前状态)字段"${h}"将被忽略`);
    }
    occurrence[h]++;
  }
  
  return warnings;
}

// 方案2: 严格模式（推荐生产环境使用）
function buildRawDataByGroupStrict(
  tableType: 1 | 2,
  headers: string[],
  rowValues: unknown[],
  expectedGroups: number[]
): Record<string, Record<string, unknown>> {
  // 验证列数与预期分组数匹配
  if (headers.length !== expectedGroups.length) {
    throw new Error(`列数不匹配: 预期${expectedGroups.length}, 实际${headers.length}`);
  }
  
  const byGroup: Record<string, Record<string, unknown>> = {};
  for (let j = 0; j < headers.length; j++) {
    const h = String(headers[j] || '').trim();
    if (!h) continue;
    
    const gid = expectedGroups[j]; // 使用预期分组，不自动计算
    if (gid === 0 || gid === 7) continue;
    
    const gKey = String(gid);
    if (!byGroup[gKey]) byGroup[gKey] = {};
    byGroup[gKey][h] = rowValues[j];
  }
  return byGroup;
}
```

```typescript
// 增加时间顺序验证
private validateTimeSequence(
  containerNumber: string,
  events: { type: string; time: Date | null }[]
): string[] {
  const errors: string[] = [];
  
  const sorted = events
    .filter(e => e.time)
    .sort((a, b) => a.time!.getTime() - b.time!.getTime());
  
  for (let i = 1; i < sorted.length; i++) {
    const prev = sorted[i - 1];
    const curr = sorted[i];
    
    // 业务规则验证
    if (curr.type === 'pickup' && prev.type === 'arrival') {
      errors.push(`时间顺序错误: 提柜时间(${curr.time})早于到港时间(${prev.time})`);
    }
    if (curr.type === 'discharge' && prev.type === 'arrival') {
      errors.push(`时间顺序错误: 卸船时间(${curr.time})早于到港时间(${prev.time})`);
    }
    if (curr.type === 'return' && prev.type === 'pickup') {
      errors.push(`时间顺序错误: 还箱时间(${curr.time})早于提柜时间(${prev.time})`);
    }
  }
  
  return errors;
}
```
